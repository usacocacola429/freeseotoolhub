function cleanDomain(input) {
// Remove protocol and www
let domain = input.replace(/^(https?:\/\/)?(www\.)?/, '');
// Remove trailing slashes
domain = domain.replace(/\/.*$/, '');
// Extract domain only (no paths)
domain = domain.split('/')[0];
return domain.toLowerCase();
}

function getDomainData(domain) {
// Check if domain is in our database
if (domainDatabase[domain]) {
return domainDatabase[domain];
}

// Generate realistic random data for unknown domains
const randomDA = Math.floor(Math.random() * 60) + 20; // 20-80
const randomPA = Math.min(randomDA + Math.floor(Math.random() * 10) - 5, 100);

const ages = ['1-2 years', '2-5 years', '5-10 years', '10+ years'];
const sslStatuses = ['Valid HTTPS', 'Invalid SSL', 'No SSL', 'Expired SSL'];
const speedScores = [65, 70, 75, 80, 85, 90];

return {
da: randomDA,
pa: randomPA,
age: ages[Math.floor(Math.random() * ages.length)],
ssl: sslStatuses[Math.floor(Math.random() * sslStatuses.length)],
speed: speedScores[Math.floor(Math.random() * speedScores.length)]
};
}
