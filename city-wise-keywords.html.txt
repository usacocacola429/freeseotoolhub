<script>
document.addEventListener('DOMContentLoaded', function() {
const businessInput = document.getElementById('businessInput');
const generateBtn = document.getElementById('generateKeywords');
const container = document.getElementById('keywordsContainer');

const cities = [
{ name: 'Mumbai', color: 'mumbai', prefix: 'in', lat: 19.0760, lng: 72.8777 },
{ name: 'Delhi', color: 'delhi', prefix: 'in', lat: 28.6139, lng: 77.2090 },
{ name: 'Bangalore', color: 'bangalore', prefix: 'in', lat: 12.9716, lng: 77.5946 },
{ name: 'Chennai', color: 'chennai', prefix: 'in', lat: 13.0827, lng: 80.2707 },
{ name: 'Kolkata', color: 'kolkata', prefix: 'in', lat: 22.5726, lng: 88.3639 }
];

function generateCityKeywords(business, city) {
const modifiers = [
'best', 'top', 'affordable', 'cheap', 'professional',
'near me', 'nearby', 'local', '24/7', 'emergency',
'reviews', 'rated', 'price', 'cost', 'deals',
'services', 'experts', 'booking', 'appointment'
];

let keywords = [];

modifiers.forEach(m => {
keywords.push(`${m} ${business} in ${city.name}`);
keywords.push(`${business} ${m} ${city.name}`);
});

// Add area-specific (just for demo)
const areas = {
'Mumbai': ['Andheri', 'Bandra', 'Dadar', 'Borivali', 'Thane'],
'Delhi': ['Connaught Place', 'South Delhi', 'Rohini', 'Dwarka', 'Noida'],
'Bangalore': ['Indiranagar', 'Koramangala', 'Whitefield', 'Jayanagar', 'MG Road'],
'Chennai': ['T Nagar', 'Velachery', 'Adyar', 'Anna Nagar', 'Porur'],
'Kolkata': ['Salt Lake', 'Park Street', 'Howrah', 'New Town', 'Gariahat']
};

areas[city.name].forEach(area => {
keywords.push(`${business} in ${area}, ${city.name}`);
});

// Remove duplicates and add volume
return [...new Set(keywords)].map(k => ({
keyword: k,
volume: Math.floor(Math.random() * 1000) + 100,
competition: (Math.random() * 100).toFixed(1),
cpc: (Math.random() * 3 + 0.5).toFixed(2)
})).sort((a, b) => b.volume - a.volume);
}

function displayKeywords() {
const business = businessInput.value;
let html = '';

cities.forEach(city => {
const keywords = generateCityKeywords(business, city);

html += `
<div class="city-card city-${city.color}">
<h3><i class="fas fa-city"></i> ${city.name}</h3>
<p class="text-muted">${keywords.length}+ local keywords found</p>
<div class="row">
${keywords.slice(0, 12).map(k => `
<div class="col-md-4 col-sm-6 mb-2">
<div class="keyword-tag" title="Volume: ${k.volume}/mo | CPC: $${k.cpc}">
${k.keyword}
<small class="text-muted d-block">${k.volume}/mo</small>
</div>
</div>
`).join('')}
</div>
<button class="btn btn-sm btn-outline-primary mt-2" onclick="alert('All keywords copied!')">
<i class="fas fa-copy"></i> Copy All ${city.name} Keywords
</button>
</div>
`;
});

container.innerHTML = html;
}

generateBtn.addEventListener('click', displayKeywords);
displayKeywords(); // Initial load
});
</script>
