<script>
// DOM Elements
const contentInput = document.getElementById('contentInput');
const categorySelect = document.getElementById('categorySelect');
const platformButtons = document.querySelectorAll('.platform-btn');
const generateBtn = document.getElementById('generateBtn');
const loader = document.getElementById('loader');
const hashtagContainer = document.getElementById('hashtagContainer');
const totalHashtags = document.getElementById('totalHashtags');
const popularityScore = document.getElementById('popularityScore');
const copyAllBtn = document.getElementById('copyAllBtn');
const clearBtn = document.getElementById('clearBtn');
const saveBtn = document.getElementById('saveBtn');
const successMessage = document.getElementById('successMessage');
const successText = document.getElementById('successText');

let currentPlatform = 'instagram';
let generatedHashtags = [];

// Platform selection
platformButtons.forEach(btn => {
btn.addEventListener('click', () => {
platformButtons.forEach(b => b.classList.remove('active'));
btn.classList.add('active');
currentPlatform = btn.dataset.platform;
});
});

// Generate Hashtags
generateBtn.addEventListener('click', generateHashtags);

// Copy All Hashtags
copyAllBtn.addEventListener('click', copyAllHashtags);

// Clear
clearBtn.addEventListener('click', clearAll);

// Save
saveBtn.addEventListener('click', saveHashtags);

function generateHashtags() {
const content = contentInput.value.trim();
const category = categorySelect.value;

if (!content && !category) {
showMessage('कृपया कंटेंट या कैटेगरी डालें!', 'error');
return;
}

// Show loader
loader.style.display = 'block';
hashtagContainer.innerHTML = '';

// Simulate AI processing
setTimeout(() => {
const hashtags = generateHashtagData(content, category, currentPlatform);
displayHashtags(hashtags);
loader.style.display = 'none';
showMessage('हैशटैग्स सफलतापूर्वक जेनरेट हो गए!', 'success');
}, 1500);
}

function generateHashtagData(content, category, platform) {
const hashtagDatabase = {
fashion: [
'#fashion', '#style', '#ootd', '#fashionblogger', '#streetstyle',
'#mensfashion', '#womensfashion', '#fashionista', '#trendy', '#luxury'
],
food: [
'#food', '#foodie', '#foodporn', '#instafood', '#yummy',
'#delicious', '#homemade', '#cooking', '#recipe', '#healthyfood'
],
travel: [
'#travel', '#wanderlust', '#travelgram', '#vacation', '#adventure',
'#explore', '#nature', '#photography', '#beautiful', '#landscape'
],
fitness: [
'#fitness', '#workout', '#gym', '#fit', '#health',
'#exercise', '#motivation', '#fitspo', '#bodybuilding', '#yoga'
],
technology: [
'#technology', '#tech', '#innovation', '#gadgets', '#ai',
'#programming', '#coding', '#software', '#digital', '#future'
]
};

// Base hashtags
let hashtags = [];

// Add category specific hashtags
if (category && hashtagDatabase[category]) {
hashtags = hashtags.concat(hashtagDatabase[category]);
}

// Add content-based hashtags (simulated AI)
const words = content.toLowerCase().split(' ').filter(word => word.length > 3);
words.forEach(word => {
hashtags.push(`#${word}`);
});

// Add platform-specific hashtags
if (platform === 'instagram') {
hashtags.push('#instagood', '#instagram', '#instadaily', '#instalike');
} else if (platform === 'twitter') {
hashtags.push('#twitter', '#tweet', '#trending', '#viral');
} else if (platform === 'facebook') {
hashtags.push('#facebook', '#fb', '#share', '#community');
}

// Add trending hashtags
hashtags.push('#trending', '#viral', '#popular', '#hot');

// Remove duplicates and limit
hashtags = [...new Set(hashtags)];

// Limit based on platform
let limit = 30;
if (platform === 'twitter') limit = 10;
if (platform === 'facebook') limit = 5;

return hashtags.slice(0, limit);
}

function displayHashtags(hashtags) {
generatedHashtags = hashtags;

// Update stats
totalHashtags.textContent = hashtags.length;
popularityScore.textContent = `${Math.min(hashtags.length * 3, 100)}%`;

// Create hashtag groups
hashtagContainer.innerHTML = '';

const groups = [
{ title: 'Popular Hashtags', hashtags: hashtags.slice(0, 10) },
{ title: 'Niche Hashtags', hashtags: hashtags.slice(10, 20) },
{ title: 'Branded Hashtags', hashtags: hashtags.slice(20) }
];

groups.forEach(group => {
if (group.hashtags.length > 0) {
const groupElement = document.createElement('div');
groupElement.className = 'hashtag-group';

let html = `<h4><i class="fas fa-tags"></i> ${group.title}</h4>`;
html += '<div class="hashtag-list">';

group.hashtags.forEach(tag => {
html += `<div class="hashtag-item" data-hashtag="${tag}">${tag}</div>`;
});

html += '</div>';
groupElement.innerHTML = html;
hashtagContainer.appendChild(groupElement);
}
});

// Add click event to each hashtag
document.querySelectorAll('.hashtag-item').forEach(item => {
item.addEventListener('click', function() {
const hashtag = this.dataset.hashtag;
copyToClipboard(hashtag);
this.classList.add('copied');
this.innerHTML = `<i class="fas fa-check"></i> ${hashtag}`;

setTimeout(() => {
this.classList.remove('copied');
this.textContent = hashtag;
}, 2000);
});
});
}

function copyAllHashtags() {
if (generatedHashtags.length === 0) {
showMessage('पहले हैशटैग्स जेनरेट करें!', 'error');
return;
}

const allHashtags = generatedHashtags.join(' ');
copyToClipboard(allHashtags);
showMessage('सभी हैशटैग्स कॉपी हो गए!', 'success');
}

function clearAll() {
contentInput.value = '';
categorySelect.value = '';
hashtagContainer.innerHTML = `
<div style="text-align: center; padding: 50px 20px; color: #94a3b8;">
<i class="fas fa-hashtag" style="font-size: 3rem; margin-bottom: 20px; opacity: 0.3;"></i>
<h3>हैशटैग्स यहाँ दिखाई देंगे</h3>
<p>ऊपर कंटेंट डालकर जेनरेट बटन दबाएं</p>
</div>
`;
totalHashtags.textContent = '0';
popularityScore.textContent = '0%';
showMessage('सभी डेटा रीसेट किया गया!', 'success');
}

function saveHashtags() {
if (generatedHashtags.length === 0) {
showMessage('पहले हैशटैग्स जेनरेट करें!', 'error');
return;
}

const data = {
content: contentInput.value,
category: categorySelect.value,
platform: currentPlatform,
hashtags: generatedHashtags,
date: new Date().toLocaleString()
};

localStorage.setItem('savedHashtags', JSON.stringify(data));
showMessage('हैशटैग्स सेव हो गए!', 'success');
}

function copyToClipboard(text) {
navigator.clipboard.writeText(text).then(() => {
console.log('Copied to clipboard:', text);
});
}

function showMessage(message, type) {
successText.textContent = message;

if (type === 'error') {
successMessage.style.background = '#dc2626';
} else {
successMessage.style.background = '#10b981';
}

successMessage.style.display = 'flex';

setTimeout(() => {
successMessage.style.display = 'none';
}, 3000);
}

// Load saved hashtags if any
window.addEventListener('load', () => {
const saved = localStorage.getItem('savedHashtags');
if (saved) {
const data = JSON.parse(saved);
contentInput.value = data.content;
categorySelect.value = data.category;

// Set platform
platformButtons.forEach(btn => {
btn.classList.remove('active');
if (btn.dataset.platform === data.platform) {
btn.classList.add('active');
currentPlatform = data.platform;
}
});

displayHashtags(data.hashtags);
}
});
