<script>
// DOM Elements
const inputUrls = document.getElementById('input-urls');
const outputContainer = document.getElementById('output-container');
const encodeBtn = document.getElementById('encode-btn');
const decodeBtn = document.getElementById('decode-btn');
const clearBtn = document.getElementById('clear-btn');
const copyBtn = document.getElementById('copy-btn');
const loader = document.getElementById('loader');
const successMessage = document.getElementById('success-message');
const inputCount = document.getElementById('input-count');
const outputCount = document.getElementById('output-count');

// Update input URL count
inputUrls.addEventListener('input', updateCounts);

// Encode URLs
encodeBtn.addEventListener('click', () => processUrls('encode'));

// Decode URLs
decodeBtn.addEventListener('click', () => processUrls('decode'));

// Clear all
clearBtn.addEventListener('click', clearAll);

// Copy results
copyBtn.addEventListener('click', copyResults);

// Update counts
function updateCounts() {
const urls = inputUrls.value.trim().split('\n').filter(url => url.trim() !== '');
inputCount.textContent = urls.length;
}

// Process URLs
function processUrls(action) {
const urls = inputUrls.value.trim().split('\n')
.map(url => url.trim())
.filter(url => url !== '');

if (urls.length === 0) {
showMessage('Please enter some URLs to process', 'error');
return;
}

// Show loader
loader.style.display = 'block';
outputContainer.innerHTML = '';

// Process in chunks to prevent blocking
setTimeout(() => {
const results = [];
let errorCount = 0;

urls.forEach((url, index) => {
try {
let processedUrl;
if (action === 'encode') {
processedUrl = encodeURIComponent(url);
} else {
processedUrl = decodeURIComponent(url);
}

results.push({
original: url,
processed: processedUrl,
success: true
});
} catch (error) {
results.push({
original: url,
processed: `Error: Invalid URL format`,
success: false
});
errorCount++;
}
});

displayResults(results, action);
loader.style.display = 'none';

const message = action === 'encode' ? 'encoded' : 'decoded';
showMessage(`Successfully ${message} ${urls.length} URL${urls.length > 1 ? 's' : ''}${errorCount > 0 ? ` (${errorCount} errors)` : ''}`);

}, 100);
}

// Display results
function displayResults(results, action) {
outputContainer.innerHTML = '';
outputCount.textContent = results.length;

if (results.length === 0) {
outputContainer.innerHTML = '<div class="url-placeholder">No results to display</div>';
return;
}

results.forEach((result, index) => {
const urlItem = document.createElement('div');
urlItem.className = `url-item ${action} ${result.success ? '' : 'error'}`;

if (!result.success) {
urlItem.innerHTML = `
<strong>‚ùå Error:</strong><br>
<span style="color: #718096;">Original: ${escapeHtml(result.original)}</span><br>
<span style="color: #f56565;">${result.processed}</span>
`;
} else {
urlItem.innerHTML = `
<strong>${action === 'encode' ? 'üîí Encoded' : 'üîì Decoded'}:</strong><br>
<span style="color: #4a5568;">${escapeHtml(result.processed)}</span>
<div style="margin-top: 5px; font-size: 0.85em; color: #a0aec0;">
Original: ${escapeHtml(result.original)}
</div>
`;
}

// Add copy button for each URL
const copyBtn = document.createElement('button');
copyBtn.innerHTML = 'üìã';
copyBtn.style.cssText = `
margin-left: 10px;
padding: 2px 8px;
background: #edf2f7;
border: none;
border-radius: 4px;
cursor: pointer;
float: right;
font-size: 12px;
`;
copyBtn.title = 'Copy this URL';
copyBtn.onclick = (e) => {
e.stopPropagation();
copyToClipboard(result.success ? result.processed : result.original);
showMessage('URL copied to clipboard!', 'success');
};

urlItem.appendChild(copyBtn);
outputContainer.appendChild(urlItem);
});
}

// Clear all
function clearAll() {
inputUrls.value = '';
outputContainer.innerHTML = '<div class="url-placeholder">Processed URLs will appear here...</div>';
inputCount.textContent = '0';
outputCount.textContent = '0';
successMessage.style.display = 'none';
showMessage('All inputs cleared', 'info');
}

// Copy results to clipboard
function copyResults() {
const urlItems = outputContainer.querySelectorAll('.url-item');
if (urlItems.length === 0) {
showMessage('No results to copy', 'error');
return;
}

const urls = Array.from(urlItems).map(item => {
const text = item.textContent.replace('üìã', '').trim();
const lines = text.split('\n');
return lines.length > 1 ? lines[1] : lines[0];
}).join('\n');

copyToClipboard(urls);
showMessage('All processed URLs copied to clipboard!', 'success');
}

// Copy text to clipboard
function copyToClipboard(text) {
navigator.clipboard.writeText(text).catch(() => {
// Fallback for older browsers
const textarea = document.createElement('textarea');
textarea.value = text;
document.body.appendChild(textarea);
textarea.select();
document.execCommand('copy');
document.body.removeChild(textarea);
});
}

// Show message
function showMessage(message, type = 'success') {
successMessage.textContent = message;
successMessage.style.display = 'block';
successMessage.style.background = type === 'error' ? '#fed7d7' :
type === 'info' ? '#bee3f8' : '#c6f6d5';
successMessage.style.color = type === 'error' ? '#9b2c2c' :
type === 'info' ? '#2c5282' : '#22543d';

setTimeout(() => {
successMessage.style.display = 'none';
}, 3000);
}

// Escape HTML
function escapeHtml(text) {
const div = document.createElement('div');
div.textContent = text;
return div.innerHTML;
}

// Initialize with example
window.addEventListener('DOMContentLoaded', () => {
const exampleUrls = `https://example.com/page with spaces.html
https://example.com/search?q=query with spaces&lang=en
https://example.com/price=‚Çπ1000&currency=INR
https://example.com/path/with/special@chars#section

http%3A%2F%2Fexample.com%2F%C3%A9%2F%C3%A7%2F%C3%A0.html`;

inputUrls.value = exampleUrls;
updateCounts();

// Auto-encode on page load for demo
setTimeout(() => processUrls('encode'), 500);
});

// Keyboard shortcuts
document.addEventListener('keydown', (e) => {
// Ctrl/Cmd + E to encode
if ((e.ctrlKey || e.metaKey) && e.key === 'e') {
e.preventDefault();
encodeBtn.click();
}
// Ctrl/Cmd + D to decode
if ((e.ctrlKey || e.metaKey) && e.key === 'd') {
e.preventDefault();
decodeBtn.click();
}
// Ctrl/Cmd + C to copy
if ((e.ctrlKey || e.metaKey) && e.key === 'c' && e.shiftKey) {
e.preventDefault();
copyBtn.click();
}
});
