function searchProducts() {
const niche = document.getElementById('productNiche').value.trim().toLowerCase();
const minPrice = parseFloat(document.getElementById('minPrice').value) || 0;
const maxPrice = parseFloat(document.getElementById('maxPrice').value) || 1000;
const category = document.getElementById('productCategory').value;
const competition = document.getElementById('competitionLevel').value;
const minSales = parseInt(document.getElementById('minSales').value) || 0;

// Show loading
document.getElementById('loadingSpinner').style.display = 'block';
document.getElementById('resultsSection').style.display = 'none';
document.getElementById('noResults').style.display = 'none';
document.getElementById('searchBtn').disabled = true;

// Simulate API call delay
setTimeout(() => {
// Filter products
let filtered = sampleProducts.filter(product => {
// Price filter
if (product.price < minPrice || product.price > maxPrice) return false;

// Category filter
if (category !== 'all' && product.category !== category) return false;

// Competition filter
if (competition !== 'any' && product.competition.toLowerCase() !== competition) return false;

// Sales filter
if (product.monthlySales < minSales) return false;

// Keyword/niche filter (simple contains check)
if (niche && !product.keywords.some(k => k.includes(niche))) return false;

return true;
});

// Sort by opportunity score (highest first)
filtered.sort((a, b) => b.opportunityScore - a.opportunityScore);

// Hide loading
document.getElementById('loadingSpinner').style.display = 'none';
document.getElementById('searchBtn').disabled = false;

if (filtered.length === 0) {
document.getElementById('noResults').style.display = 'block';
document.getElementById('resultsSection').style.display = 'none';
return;
}

// Update summary stats
document.getElementById('totalProducts').textContent = filtered.length;
const avgPrice = filtered.reduce((sum, p) => sum + p.price, 0) / filtered.length;
document.getElementById('avgPrice').textContent = '$' + avgPrice.toFixed(2);
const avgSales = filtered.reduce((sum, p) => sum + p.monthlySales, 0) / filtered.length;
document.getElementById('avgSales').textContent = Math.round(avgSales);
const avgRating = filtered.reduce((sum, p) => sum + p.rating, 0) / filtered.length;
document.getElementById('avgRating').textContent = avgRating.toFixed(1);

// Display products
displayProducts(filtered);

// Update charts
updateCharts(filtered);

// Show results
document.getElementById('resultsSection').style.display = 'block';
}, 1500);
}
