(function() {
// DOM Elements
const urlInput = document.getElementById('urlInput');

const clsRating = document.getElementById('clsRating');

const insightsGrid = document.getElementById('insightsGrid');

// Chart instance
let vitalsChart;
let currentRange = '12m';
let currentUrl = 'https://freeseotoolhub.com';

// Generate realistic historical data based on URL and range
function generateHistoricalData(url, range) {
// Seed based on URL to get consistent data per URL
const urlSeed = url.split('').reduce((acc, char) => acc + char.charCodeAt(0), 0);

// Number of data points based on range
let points = [];
let months = range === '12m' ? 12 : range === '6m' ? 6 : range === '3m' ? 3 : 1;

// For 30 days, use daily data
if (range === '30d') {
months = 30; // 30 days
}

const now = new Date();

for (let i = months; i >= 0; i--) {
let date;
if (range === '30d') {
date = new Date(now);
date.setDate(date.getDate() - i);
} else {
date = new Date(now);
date.setMonth(date.getMonth() - i);
}

// Generate realistic values with some randomness but consistent trend
// LCP: between 1.8 and 4.5 seconds
const lcpBase = 2.2 + (urlSeed % 20) / 10;
const lcpTrend = Math.sin(i / 3) * 0.4; // Cyclical trend
const lcpRandom = (Math.sin(i * urlSeed) * 0.2);
let lcp = lcpBase + lcpTrend + lcpRandom;
lcp = Math.max(1.5, Math.min(5.5, lcp));

// INP: between 120 and 450 ms
const inpBase = 180 + (urlSeed % 30);
const inpTrend = Math.cos(i / 2.5) * 30;
const inpRandom = (Math.cos(i * urlSeed) * 20);
let inp = inpBase + inpTrend + inpRandom;
inp = Math.max(100, Math.min(550, inp));

// CLS: between 0.02 and 0.3
const clsBase = 0.06 + (urlSeed % 5) / 100;
const clsTrend = Math.sin(i / 4) * 0.04;
const clsRandom = (Math.sin(i * urlSeed * 2) * 0.02);
let cls = clsBase + clsTrend + clsRandom;
cls = Math.max(0.01, Math.min(0.4, cls));

points.push({
date: date.toLocaleDateString('en-US', range === '30d' ? { month: 'short', day: 'numeric' } : { month: 'short', year: 'numeric' }),
lcp: parseFloat(lcp.toFixed(2)),
inp: Math.round(inp),
cls: parseFloat(cls.toFixed(3))
});
}

return points;
}

// Get rating based on value and metric
function getRating(metric, value) {
if (metric === 'lcp') {
if (value <= 2.5) return { text: 'Good', class: 'rating-good' };
if (value <= 4.0) return { text: 'Needs Improvement', class: 'rating-needs-improvement' };
return { text: 'Poor', class: 'rating-poor' };
} else if (metric === 'inp') {
if (value <= 200) return { text: 'Good', class: 'rating-good' };
if (value <= 500) return { text: 'Needs Improvement', class: 'rating-needs-improvement' };
return { text: 'Poor', class: 'rating-poor' };
} else if (metric === 'cls') {
if (value <= 0.1) return { text: 'Good', class: 'rating-good' };
if (value <= 0.25) return { text: 'Needs Improvement', class: 'rating-needs-improvement' };
return { text: 'Poor', class: 'rating-poor' };
}
}

// Calculate trend between first and last point
function getTrend(data, metric) {
if (data.length < 2) return 'stable';
const first = data[0][metric];
const last = data[data.length - 1][metric];
const diff = last - first;

if (metric === 'cls') {
if (diff < -0.01) return 'improving';
if (diff > 0.01) return 'degrading';
return 'stable';
} else {
// LCP and INP: lower is better
if (diff < -0.2) return 'improving';
if (diff > 0.2) return 'degrading';
return 'stable';
}
}

// Update the UI with new data
function updateVitalsDisplay(url) {
const historicalData = generateHistoricalData(url, currentRange);
const latest = historicalData[historicalData.length - 1];

// Update current values
lcpValue.textContent = latest.lcp;
inpValue.textContent = latest.inp;
clsValue.textContent = latest.cls.toFixed(3);

// Update ratings
const lcpRatingObj = getRating('lcp', latest.lcp);
const inpRatingObj = getRating('inp', latest.inp);
const clsRatingObj = getRating('cls', latest.cls);

lcpRating.textContent = lcpRatingObj.text;
lcpRating.className = `vital-rating ${lcpRatingObj.class}`;

inpRating.textContent = inpRatingObj.text;
inpRating.className = `vital-rating ${inpRatingObj.class}`;

clsRating.textContent = clsRatingObj.text;
clsRating.className = `vital-rating ${clsRatingObj.class}`;

// Update history table
let tableHtml = '';
historicalData.slice().reverse().forEach((item, index) => {
// Calculate trend for each row (compared to previous)
let trendIcon = '➡️';
let trendClass = 'trend-stable';

if (index < historicalData.length - 1) {
const prev = historicalData[historicalData.length - index - 2];
const lcpDiff = prev.lcp - item.lcp;
if (Math.abs(lcpDiff) > 0.1) {
if (lcpDiff > 0) {
trendIcon = '📈';
trendClass = 'trend-up';
} else {
trendIcon = '📉';
trendClass = 'trend-down';
}
}
}

tableHtml += `
<tr>
<td><strong>${item.date}</strong></td>
<td>${item.lcp}s</td>
<td>${item.inp}ms</td>
<td>${item.cls}</td>
<td><span class="trend-indicator ${trendClass}">${trendIcon} </span></td>
</tr>
`;
});

historyBody.innerHTML = tableHtml;

// Update insights based on trends
const lcpTrend = getTrend(historicalData, 'lcp');
const inpTrend = getTrend(historicalData, 'inp');
const clsTrend = getTrend(historicalData, 'cls');

let insightsHtml = '';

if (lcpTrend === 'improving') {
insightsHtml += `<div class="insight-item"><strong>✅ LCP is improving</strong><p>Largest Contentful Paint has decreased by ${(historicalData[0].lcp - latest.lcp).toFixed(2)}s over the period. Keep optimizing!</p></div>`;
} else if (lcpTrend === 'degrading') {
insightsHtml += `<div class="insight-item"><strong>⚠️ LCP is degrading</strong><p>LCP increased by ${(latest.lcp - historicalData[0].lcp).toFixed(2)}s. Check server response time and render-blocking resources.</p></div>`;
} else {
insightsHtml += `<div class="insight-item"><strong>📊 LCP is stable</strong><p>LCP has remained consistent. Consider optimizations to push it below 2.0s for better user experience.</p></div>`;
}

if (inpTrend === 'improving') {
insightsHtml += `<div class="insight-item"><strong>✅ INP is improving</strong><p>Interaction to Next Paint improved by ${historicalData[0].inp - latest.inp}ms. Great work on JavaScript optimization!</p></div>`;
} else if (inpTrend === 'degrading') {
insightsHtml += `<div class="insight-item"><strong>⚠️ INP needs attention</strong><p>INP increased by ${latest.inp - historicalData[0].inp}ms. Look for long tasks and third-party script impact.</p></div>`;
} else {
insightsHtml += `<div class="insight-item"><strong>👆 INP is stable</strong><p>INP has stayed within a consistent range. Aim for under 200ms for all interactions.</p></div>`;
}

if (clsTrend === 'improving') {
insightsHtml += `<div class="insight-item"><strong>✅ CLS is excellent</strong><p>CLS improved by ${(historicalData[0].cls - latest.cls).toFixed(3)}. Maintain current practices with dimension attributes.</p></div>`;
} else if (clsTrend === 'degrading') {
insightsHtml += `<div class="insight-item"><strong>⚠️ CLS is increasing</strong><p>CLS increased by ${(latest.cls - historicalData[0].cls).toFixed(3)}. Check for layout shifts from ads or images without dimensions.</p></div>`;
} else {
insightsHtml += `<div class="insight-item"><strong>📐 CLS is stable</strong><p>CLS remains under control. Continue monitoring for any unexpected layout shifts.</p></div>`;
}

// Overall score
const overallScore = Math.round(100 - (latest.lcp * 3) - (latest.inp / 20) - (latest.cls * 50));
const finalScore = Math.max(60, Math.min(98, overallScore));
insightsHtml += `<div class="insight-item"><strong>📈 Overall Score: ${finalScore}/100</strong><p>Based on Core Web Vitals weighted analysis. ${finalScore > 85 ? 'Excellent!' : finalScore > 70 ? 'Good, but room for improvement.' : 'Needs significant work.'}</p></div>`;

insightsGrid.innerHTML = insightsHtml;

// Update chart
updateChart(historicalData);
}

}
