// Update chain visualization
function updateChainVisualization(data) {
let html = '';
data.chain.forEach((node, index) => {
let nodeClass = 'chain-node';
let codeClass = '';

if (node.status === 200) {
nodeClass += ' final';
codeClass = 'code-200';
} else if (node.status === 404) {
nodeClass += ' error';
codeClass = 'code-404';
} else if (node.status === 301) {
nodeClass += ' redirect';
codeClass = 'code-301';
} else if (node.status === 302) {
nodeClass += ' redirect';
codeClass = 'code-302';
}

html += `
<div class="${nodeClass}">
<div style="margin-bottom: 0.5rem;">
<span class="http-code ${codeClass}">${node.status}</span>
<span style="margin-left: 0.5rem; font-weight: 600;">${node.type}</span>
</div>
<div style="font-size: 0.9rem; word-break: break-all;">
${node.url}
</div>
${node.url.includes('LOOP') ? '<div style="color: #EF4444; margin-top: 0.5rem;">⚠️ LOOP DETECTED</div>' : ''}
</div>
`;

if (index < data.chain.length - 1) {
html += '<div class="chain-arrow">→</div>';
}
});

document.getElementById('chainFlow').innerHTML = html;
}

// Bulk URL check
function checkBulkURLs() {

urls.forEach(url => {
const data = generateRedirectChain(url.trim());

data.seoImpact === 'Warning' ? '#F59E0B' : '#EF4444';

results += `
<div class="url-item">
<span class="url-status" style="background: ${statusColor}20; color: ${statusColor};">
${data.chainLength} redirects
</span>
<span style="flex: 1;">${url}</span>
<span>→ ${data.finalUrl.substring(0, 50)}${data.finalUrl.length > 50 ? '...' : ''}</span>
</div>
`;
});

document.getElementById('bulkResults').innerHTML = results;
document.getElementById('bulkResults').style.display = 'block';
}
