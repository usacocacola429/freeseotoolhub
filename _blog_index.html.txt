// Save Post Function
function savePost() {
// Get form values
const title = document.getElementById('postTitle').value;
const category = document.getElementById('postCategory').value;
const author = document.getElementById('postAuthor').value;
const image = document.getElementById('postImage').value;
const excerpt = document.getElementById('postExcerpt').value;
const content = quill.root.innerHTML;
const keywords = document.getElementById('postKeywords').value;

// Create post object
const newPost = {
id: Date.now(),
title: title,
slug: title.toLowerCase().replace(/[^a-z0-9]+/g, '-'),
category: category,
date: new Date().toISOString().split('T')[0],
author: author,
image: image || 'default.jpg',
excerpt: excerpt,
content: content,
keywords: keywords.split(',').map(k => k.trim()),
views: 0,
comments: 0,
featured: false
};

// Get existing posts
let posts = JSON.parse(localStorage.getItem('blogPosts')) || [];
posts.unshift(newPost); // Add new post at top

// Save to localStorage
localStorage.setItem('blogPosts', JSON.stringify(posts));

alert('‚úÖ ‡§™‡•ã‡§∏‡•ç‡§ü ‡§∏‡•á‡§µ ‡§π‡•ã ‡§ó‡§à!');
toggleAdminPanel();
loadPosts();

// Clear form
document.getElementById('postForm').reset();
quill.setContents([]);
}

function updateCategories(posts) {
const cats = {};
posts.forEach(p => {
cats[p.category] = (cats[p.category] || 0) + 1;
});

let html = '';
for (let cat in cats) {
html += `<li><a href="#">${cat} <span>(${cats[cat]})</span></a></li>`;
}
document.getElementById('categoryList').innerHTML = html;
}

function updatePopularPosts(posts) {
const popular = [...posts].sort((a,b) => b.views - a.views).slice(0,5);

let html = '';
popular.forEach(p => {
html += `
<li>
<a href="post.html?id=${p.id}">${p.title}</a>
<span class="date">${p.date} ‚Ä¢ ${p.views.toLocaleString()} views</span>
</li>
`;
});
document.getElementById('popularPosts').innerHTML = html;
}

function editPost(id) {
const posts = JSON.parse(localStorage.getItem('blogPosts')) || [];
const post = posts.find(p => p.id === id);

if (post) {
document.getElementById('postTitle').value = post.title;
document.getElementById('postCategory').value = post.category;
document.getElementById('postAuthor').value = post.author;
document.getElementById('postImage').value = post.image;
document.getElementById('postExcerpt').value = post.excerpt;
document.getElementById('postKeywords').value = post.keywords.join(', ');

quill.root.innerHTML = post.content;

toggleAdminPanel();
}
}
