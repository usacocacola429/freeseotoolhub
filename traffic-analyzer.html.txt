// Export CSV
function exportCSV(data) {
const rows = [
['Metric', 'Value'],
['Domain', data.domain],
['Market', data.market.toUpperCase()],
['Period', `${data.period.start} to ${data.period.end}`],
['Total Visits', data.monthlyVisits],
['Bounce Rate', `${data.bounceRate}%`],
['Avg Session Duration', `${Math.floor(data.avgSessionDuration / 60)}:${data.avgSessionDuration % 60}`],
['Pages Per Session', data.pagesPerSession],
[''],
['Traffic Sources'],
['Source', 'Visits', 'Percentage']
];

Object.values(data.sources).forEach(s => {
rows.push([s.name, s.visits, `${s.percent}%`]);
});

rows.push([''], ['Geographic Distribution']);
rows.push(['Country', 'Visits', 'Percentage']);
data.geoData.forEach(g => {
rows.push([g.country, g.visits, `${g.percent}%`]);
});

rows.push([''], ['Device Breakdown']);
rows.push(['Device', 'Visits', 'Percentage']);
data.deviceData.forEach(d => {
rows.push([d.name, d.visits, `${d.percent}%`]);
});

const csvContent = rows.map(row => row.join(',')).join('\n');
const blob = new Blob([csvContent], { type: 'text/csv' });
const url = window.URL.createObjectURL(blob);
const a = document.createElement('a');
a.href = url;
a.download = `traffic-report-${data.domain}-${new Date().toISOString().split('T')[0]}.csv`;
a.click();
}
