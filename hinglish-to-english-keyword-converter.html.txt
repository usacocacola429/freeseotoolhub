function convertHinglishToEnglish(text) {
// Common Hinglish to English mappings
const mappings = {
'मोबाइल': 'mobile',
'खरीदें': 'buy',
'कैसे': 'how',
'अच्छा': 'good best',
'सा': '',
'कौन': 'which',
'है': 'is',
'ऑनलाइन': 'online',
'पैसे': 'money',
'कमाए': 'earn',
'सस्ता': 'cheap budget',
'फ्रिज': 'refrigerator',
'बाल': 'hair',
'कटवाने': 'cut',
'तरीका': 'way method',
'अमेज़न': 'amazon',
'सामान': 'product items',
'ऑर्डर': 'order',
'मँगवाएँ': 'order',
'बेस्ट': 'best',
'रेस्टोरेंट': 'restaurant',
'कहाँ': 'where',
'नियर': 'near',
'मी': 'me',
'लैपटॉप': 'laptop',
'फोन': 'phone',
'नंबर': 'number',
'ट्रेस': 'trace track',
'पैन': 'pan',
'कार्ड': 'card',
'बनवाएं': 'apply make',
'आधार': 'aadhaar',
'निकाले': 'withdraw',
'गूगल': 'google',
'पे': 'pay',
'ट्रांसफर': 'transfer'
};

let result = text.toLowerCase();

// Replace common patterns
result = result.replace(/कैसे/g, 'how to');
result = result.replace(/कौन सा/g, 'which');
result = result.replace(/अच्छा सा/g, 'best');
result = result.replace(/कहाँ है/g, 'near me');
result = result.replace(/से/g, 'from');

// Apply mappings
for (let [hing, eng] of Object.entries(mappings)) {
const regex = new RegExp(hing, 'gi');
result = result.replace(regex, eng);
}

// Clean up extra spaces
result = result.replace(/\s+/g, ' ').trim();

return result || 'how to buy mobile';
}
