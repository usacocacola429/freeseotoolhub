function getSERPData(keyword, website) {
// Check if keyword is in our database
if (serpDatabase[keyword]) {
return serpDatabase[keyword];
}

// Generate realistic SERP data for new keywords
const positions = [];
const positionCount = Math.floor(Math.random() * 5) + 5; // 5-10 results

for (let i = 1; i <= positionCount; i++) {
const isFeatured = i === 1 && Math.random() > 0.7; // 30% chance first is featured
const hasWebsite = website && Math.random() > 0.8; // 20% chance user's website appears

positions.push({
position: i,
url: hasWebsite ? website : generateRandomURL(),
title: generateRandomTitle(keyword),
description: generateRandomDescription(keyword),
featured: isFeatured
});
}

// Calculate average position (weighted)
const avgPos = positions.reduce((sum, pos) => sum + pos.position, 0) / positions.length;

// Determine competition level
const compLevels = ['Low', 'Medium', 'High', 'Very High'];
const competition = compLevels[Math.floor(Math.random() * compLevels.length)];

return {
positions: positions,
avgPosition: avgPos.toFixed(1),
competition: competition
};
}

function generateRandomURL() {
const domains = ['example.com', 'testwebsite.org', 'demo-site.net', 'sampleweb.app'];
const paths = ['blog', 'article', 'post', 'guide', 'tutorial'];
const domain = domains[Math.floor(Math.random() * domains.length)];
const path = paths[Math.floor(Math.random() * paths.length)];
return `https://${domain}/${path}/${Math.floor(Math.random() * 1000)}`;
}

function generateRandomTitle(keyword) {
const prefixes = ['Best', 'Complete Guide to', 'How to', 'Ultimate', 'Free'];
const suffixes = ['in 2024', 'for Beginners', 'Step by Step', 'Tips and Tricks'];
const prefix = prefixes[Math.floor(Math.random() * prefixes.length)];
const suffix = suffixes[Math.floor(Math.random() * suffixes.length)];
return `${prefix} ${keyword} ${suffix}`;
}

function generateRandomDescription(keyword) {
const templates = [

`Free resources and tools for ${keyword}. Improve your skills and get better results.`,
`Complete tutorial on ${keyword}. From basics to advanced techniques.`
];
return templates[Math.floor(Math.random() * templates.length)];
}
