<div class="loader-spinner"></div>
<p>Generating QR Code...</p>
</div>

<!-- JavaScript -->
<script>
// DOM Elements
const upiIdInput = document.getElementById('upiId');
const payeeNameInput = document.getElementById('payeeName');
const amountInput = document.getElementById('amount');
const noteInput = document.getElementById('note');
const generateBtn = document.getElementById('generateBtn');
const resetBtn = document.getElementById('resetBtn');
const downloadBtn = document.getElementById('downloadBtn');
const copyLinkBtn = document.getElementById('copyLinkBtn');
const qrcodeDiv = document.getElementById('qrcode');
const displayUpiId = document.getElementById('displayUpiId');
const displayPayee = document.getElementById('displayPayee');
const displayAmount = document.getElementById('displayAmount');
const displayNote = document.getElementById('displayNote');
const upiLinkPreview = document.getElementById('upiLinkPreview');
const upiLink = document.getElementById('upiLink');
const successMessage = document.getElementById('successMessage');
const successText = document.getElementById('successText');
const loader = document.getElementById('loader');
const appCards = document.querySelectorAll('.upi-app-card');

let qrCode = null;

// App selection
appCards.forEach(card => {
card.addEventListener('click', function() {
appCards.forEach(c => c.classList.remove('active'));
this.classList.add('active');
updatePlaceholder(this.dataset.app);
});
});

// Update placeholder based on app
function updatePlaceholder(app) {
const examples = {
'all': 'example@okhdfcbank',
'gpay': 'example@okhdfcbank',
'phonepe': '9876543210@ybl',
'paytm': '9876543210@paytm',
'bhim': '9876543210@upi',
'amazon': '9876543210@apl'
};
upiIdInput.placeholder = examples[app] || 'example@okhdfcbank';
}

// Generate UPI Link
function generateUPILink() {
let upiId = upiIdInput.value.trim();
if (!upiId) {
showMessage('Please enter UPI ID!', 'error');
return false;
}

// Validate UPI ID format (basic)
if (!upiId.includes('@')) {
showMessage('Invalid UPI ID format! Must contain @ (e.g., name@bank)', 'error');
return false;
}

const payee = payeeNameInput.value.trim() || 'Customer';
const amount = amountInput.value.trim();
const note = noteInput.value.trim() || 'Payment';

// Build UPI link
let upiLinkStr = `upi://pay?pa=${encodeURIComponent(upiId)}`;
upiLinkStr += `&pn=${encodeURIComponent(payee)}`;
if (amount) {
upiLinkStr += `&am=${amount}`;
}
upiLinkStr += `&tn=${encodeURIComponent(note)}`;
upiLinkStr += `&cu=INR`;

return upiLinkStr;
}

// Generate QR Code
function generateQR() {
loader.style.display = 'block';
qrcodeDiv.innerHTML = '';

const upiLinkStr = generateUPILink();
if (!upiLinkStr) {
loader.style.display = 'none';
return;
}

setTimeout(() => {
// Clear previous QR
qrcodeDiv.innerHTML = '';

// Generate new QR
qrCode = new QRCode(qrcodeDiv, {
text: upiLinkStr,
width: 256,
height: 256,
colorDark: "#000000",
colorLight: "#ffffff",
correctLevel: QRCode.CorrectLevel.H
});

// Update display
displayUpiId.textContent = upiIdInput.value;
displayPayee.textContent = payeeNameInput.value || 'Not specified';
displayAmount.textContent = amountInput.value || 'Any amount';
displayNote.textContent = noteInput.value || 'Not specified';

// Show UPI link
upiLink.textContent = upiLinkStr;
upiLinkPreview.style.display = 'block';

loader.style.display = 'none';
showMessage('‚úÖ QR Code generated successfully!', 'success');
}, 500);
}

// Download QR Code
function downloadQR() {
const canvas = qrcodeDiv.querySelector('canvas');
if (!canvas) {
showMessage('Generate QR code first!', 'error');
return;
}

const link = document.createElement('a');
link.download = `UPI-QR-${Date.now()}.png`;
link.href = canvas.toDataURL('image/png');
link.click();
showMessage('QR Code downloaded!', 'success');
}

// Copy UPI Link
function copyLink() {
if (!upiLink.textContent || upiLink.textContent === 'upi://pay?pa=example@okhdfcbank&pn=Free%20SEO%20Tool%20Hub') {
showMessage('Generate QR code first!', 'error');
return;
}

navigator.clipboard.writeText(upiLink.textContent).then(() => {
showMessage('UPI link copied to clipboard!', 'success');
});
}

// Reset Form
function resetForm() {
upiIdInput.value = 'freeseotoolhub@okhdfcbank';
payeeNameInput.value = 'Free SEO Tool Hub';
amountInput.value = '';
noteInput.value = 'Payment via Free SEO Tool Hub';

// Clear active app
appCards.forEach(c => c.classList.remove('active'));
document.querySelector('[data-app="all"]').classList.add('active');

// Clear QR
qrcodeDiv.innerHTML = '';
displayUpiId.textContent = '-';
displayPayee.textContent = '-';
displayAmount.textContent = '-';
displayNote.textContent = '-';
upiLinkPreview.style.display = 'none';

showMessage('Form reset to default', 'info');
}

// Show Message
function showMessage(message, type = 'success') {
successText.textContent = message;
if (type === 'error') successMessage.style.background = '#ef4444';
else if (type === 'info') successMessage.style.background = '#3b82f6';
else successMessage.style.background = '#10b981';

successMessage.style.display = 'flex';
setTimeout(() => {
successMessage.style.display = 'none';
}, 3000);
}

// Event Listeners
generateBtn.addEventListener('click', generateQR);
resetBtn.addEventListener('click', resetForm);
downloadBtn.addEventListener('click', downloadQR);
copyLinkBtn.addEventListener('click', copyLink);

// Enter key
upiIdInput.addEventListener('keypress', (e) => {
if (e.key === 'Enter') generateQR();
});

// Auto generate on page load
window.onload = function() {
setTimeout(() => {
generateQR();
}, 1000);
};

// Keyboard shortcuts
document.addEventListener('keydown', (e) => {
if (e.ctrlKey && e.key === 'g') {
e.preventDefault();
generateQR();
}
if (e.ctrlKey && e.key === 'd') {
e.preventDefault();
downloadQR();
}
});
</script>
</body>
</html>
