<script>
function analyzeContent() {
const text = document.getElementById('textInput').value;
const resultBox = document.getElementById('result');
const aiScoreSpan = document.getElementById('aiScore');
const humanScoreSpan = document.getElementById('humanScore');
const progressFill = document.getElementById('progressFill');
const verdict = document.getElementById('verdict');

if (text.length < 20) {
alert("Please enter at least 20 characters for analysis.");
return;
}

// Simulation Logic (Note: Real AI detection requires a backend API)
// This uses heuristic checks for demo purposes (repetitive patterns, perfect grammar lack of emotion)
const wordCount = text.split(' ').length;
const avgWordLength = text.replace(/[^a-zA-Z ]/g, "").length / wordCount;
const hasComplexTransitions = ["Furthermore", "Moreover", "In conclusion", "Additionally"].some(w => text.includes(w));

let aiProbability = 50; // Base

if (hasComplexTransitions) aiProbability += 20;
if (avgWordLength > 6) aiProbability += 15;
if (text.includes("?") && text.includes("!")) aiProbability -= 10; // Humans use mixed punctuation

// Randomize slightly for demo feel
aiProbability = Math.min(99, Math.max(5, aiProbability + Math.floor(Math.random() * 10)));
let humanProbability = 100 - aiProbability;

resultBox.style.display = 'block';
resultBox.className = 'result-box ' + (aiProbability > 60 ? 'ai-high' : 'human-high');

aiScoreSpan.innerText = aiProbability + "%";
humanScoreSpan.innerText = humanProbability + "%";

setTimeout(() => {
progressFill.style.width = aiProbability + "%";
progressFill.style.backgroundColor = aiProbability > 60 ? '#ff4d4d' : '#38b2ac';
}, 100);

if (aiProbability > 70) {
verdict.innerText = "⚠️ Likely AI Generated. Consider adding personal insights.";
} else if (aiProbability > 40) {
verdict.innerText = "🤔 Mixed. Could be AI-edited human content.";
} else {
verdict.innerText = "✅ Likely Human Written. Natural flow detected.";
}
}
</script>
