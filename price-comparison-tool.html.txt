function generatePriceComparison(product, stores) {
// Product database with realistic prices
const productDatabase = {
'iphone 15': {
amazon: { price: 79990, original: 89990, rating: 4.5, delivery: '1 day', emi: true },
flipkart: { price: 78999, original: 87999, rating: 4.4, delivery: '2 days', emi: true },
myntra: { price: 81990, original: 89990, rating: 4.3, delivery: '3 days', emi: false },
ajio: { price: 82990, original: 89990, rating: 4.2, delivery: '4 days', emi: true }
},
'nike shoes': {
amazon: { price: 3499, original: 4999, rating: 4.3, delivery: '2 days', emi: false },
flipkart: { price: 3299, original: 4599, rating: 4.2, delivery: '3 days', emi: false },
myntra: { price: 2999, original: 4999, rating: 4.5, delivery: '1 day', emi: false },
ajio: { price: 3199, original: 4499, rating: 4.1, delivery: '2 days', emi: false }
},
'samsung tv': {
amazon: { price: 42999, original: 49999, rating: 4.4, delivery: '3 days', emi: true },
flipkart: { price: 41999, original: 48999, rating: 4.3, delivery: '2 days', emi: true },
myntra: { price: 43999, original: 49999, rating: 4.2, delivery: '4 days', emi: false },
ajio: { price: 44999, original: 49999, rating: 4.1, delivery: '5 days', emi: true }
}
};

// Check if product exists in database
const searchTerm = product.toLowerCase();
let matchedProduct = null;

for (const [key, data] of Object.entries(productDatabase)) {
if (searchTerm.includes(key)) {
matchedProduct = data;
break;
}
}

// If not found, generate random prices
if (!matchedProduct) {
matchedProduct = {};
stores.forEach(store => {
const basePrice = Math.floor(Math.random() * 50000) + 1000;
matchedProduct[store] = {
price: basePrice,
original: Math.floor(basePrice * 1.2),
rating: (Math.random() * 1.5 + 3.5).toFixed(1),
delivery: Math.floor(Math.random() * 4) + 1 + ' days',
emi: Math.random() > 0.3
};
});
}

// Filter by selected stores
const results = [];
let lowestPrice = Infinity;
let bestStore = '';

stores.forEach(store => {
if (matchedProduct[store]) {
const data = matchedProduct[store];
results.push({
store: store,
name: product,
price: data.price,
originalPrice: data.original,
rating: data.rating,
delivery: data.delivery,
emi: data.emi,
discount: Math.round((1 - data.price / data.original) * 100)
});

if (data.price < lowestPrice) {
lowestPrice = data.price;
bestStore = store;
}
}
});

// Mark best deal
results.forEach(result => {
result.isBest = result.store === bestStore;
});

return results;
}

function getStoreIcon(store) {
const icons = {
'amazon': '<i class="fab fa-amazon"></i>',
'flipkart': '<i class="fas fa-shopping-cart"></i>',
'myntra': '<i class="fas fa-tshirt"></i>',
'ajio': '<i class="fas fa-shopping-bag"></i>'
};
return icons[store] || '<i class="fas fa-store"></i>';
}

function getStoreName(store) {
const names = {
'amazon': 'Amazon',
'flipkart': 'Flipkart',
'myntra': 'Myntra',
'ajio': 'Ajio'
};
return names[store] || store;
}

function saveForLater(store, price) {
const savedItems = JSON.parse(localStorage.getItem('savedProducts') || '[]');
savedItems.push({
store: store,
price: price,
date: new Date().toLocaleString()
});
localStorage.setItem('savedProducts', JSON.stringify(savedItems));
showMessage('प्रोडक्ट सेव हो गया!', 'success');
}
