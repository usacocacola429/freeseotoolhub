<script>
// Global variables
let originalFile = null;
let originalImage = null;
let compressedImageUrl = null;
let compressedBlob = null;
let originalSize = 0;

// DOM Elements
const uploadArea = document.getElementById('uploadArea');
const fileInput = document.getElementById('fileInput');
const fileInfo = document.getElementById('fileInfo');
const compressionControls = document.getElementById('compressionControls');
const previewSection = document.getElementById('previewSection');
const actionButtons = document.getElementById('actionButtons');
const loadingSpinner = document.getElementById('loadingSpinner');
const qualitySlider = document.getElementById('qualitySlider');
const qualityValue = document.getElementById('qualityValue');
const compressBtn = document.getElementById('compressBtn');
const downloadBtn = document.getElementById('downloadBtn');
const resetBtn = document.getElementById('resetBtn');
const notification = document.getElementById('notification');

// Preview elements
const originalContainer = document.getElementById('originalContainer');
const compressedContainer = document.getElementById('compressedContainer');
const originalName = document.getElementById('originalName');
const originalSizeEl = document.getElementById('originalSize');
const originalDimensions = document.getElementById('originalDimensions');
const compressedFormat = document.getElementById('compressedFormat');
const compressedSize = document.getElementById('compressedSize');
const savings = document.getElementById('savings');
const savingsBadge = document.getElementById('savingsBadge');
const savePercent = document.getElementById('savePercent');

// Event Listeners
uploadArea.addEventListener('click', () => fileInput.click());

uploadArea.addEventListener('dragover', (e) => {
e.preventDefault();
uploadArea.style.borderColor = '#667eea';
uploadArea.style.background = '#f0f4ff';
});

uploadArea.addEventListener('dragleave', () => {
uploadArea.style.borderColor = '#cbd5e0';
uploadArea.style.background = '#f8fafc';
});

uploadArea.addEventListener('drop', (e) => {
e.preventDefault();
uploadArea.style.borderColor = '#cbd5e0';
uploadArea.style.background = '#f8fafc';

const file = e.dataTransfer.files[0];
if (file && file.type.startsWith('image/')) {
handleFileSelect(file);
} else {
showNotification('Please drop an image file!', 'error');
}
});

fileInput.addEventListener('change', (e) => {
if (e.target.files[0]) {
handleFileSelect(e.target.files[0]);
}
});

qualitySlider.addEventListener('input', (e) => {
qualityValue.textContent = e.target.value + '%';
});

compressBtn.addEventListener('click', compressImage);
downloadBtn.addEventListener('click', downloadImage);
resetBtn.addEventListener('click', resetTool);

// Handle file selection
function handleFileSelect(file) {
originalFile = file;
originalSize = file.size;

// Update file info
fileInfo.textContent = `Selected: ${file.name} (${formatBytes(file.size)})`;

// Show compression controls
compressionControls.style.display = 'block';

// Display original image
const reader = new FileReader();
reader.onload = (e) => {
originalImage = new Image();
originalImage.src = e.target.result;
originalImage.onload = () => {
originalContainer.innerHTML = `<img src="${e.target.result}" alt="Original">`;
originalName.textContent = file.name;
originalSizeEl.textContent = formatBytes(file.size);
originalDimensions.textContent = `${originalImage.width} x ${originalImage.height}`;

// Hide preview section initially
previewSection.style.display = 'none';
actionButtons.style.display = 'block';
};
};
reader.readAsDataURL(file);
}

// Compress image
function compressImage() {
if (!originalImage) {
showNotification('Please select an image first!', 'error');
return;
}

loadingSpinner.style.display = 'block';
compressBtn.disabled = true;

setTimeout(() => {
const quality = qualitySlider.value / 100;
const format = document.querySelector('input[name="format"]:checked').value;

// Create canvas
const canvas = document.createElement('canvas');
const ctx = canvas.getContext('2d');

// Calculate new dimensions (maintain aspect ratio)
let width = originalImage.width;
let height = originalImage.height;

// Max dimension 1920px (for performance)
const maxDimension = 1920;
if (width > maxDimension || height > maxDimension) {
if (width > height) {
height = Math.round((height * maxDimension) / width);
width = maxDimension;
} else {
width = Math.round((width * maxDimension) / height);
height = maxDimension;
}
}

canvas.width = width;
canvas.height = height;

// Draw image
ctx.drawImage(originalImage, 0, 0, width, height);

// Convert to blob
let mimeType = 'image/jpeg';
if (format === 'png') mimeType = 'image/png';
if (format === 'webp') mimeType = 'image/webp';

canvas.toBlob((blob) => {
compressedBlob = blob;
const compressedUrl = URL.createObjectURL(blob);
compressedImageUrl = compressedUrl;

// Display compressed image
compressedContainer.innerHTML = `<img src="${compressedUrl}" alt="Compressed">`;

// Update stats
const formatNames = {
'jpeg': 'JPEG',
'png': 'PNG',
'webp': 'WebP'
};
compressedFormat.textContent = formatNames[format];
compressedSize.textContent = formatBytes(blob.size);

const savingsPercent = Math.round(((originalSize - blob.size) / originalSize) * 100);
savings.textContent = formatBytes(originalSize - blob.size) + ' (' + savingsPercent + '%)';

if (savingsPercent > 0) {
savePercent.textContent = savingsPercent + '%';
savingsBadge.style.display = 'inline-block';
} else {
savingsBadge.style.display = 'none';
}

// Show preview section
previewSection.style.display = 'grid';

// Hide loading
loadingSpinner.style.display = 'none';
compressBtn.disabled = false;

showNotification('Image compressed successfully!', 'success');

}, mimeType, quality);

}, 1000);
}

// Download compressed image
function downloadImage() {
if (!compressedBlob) {
showNotification('Please compress an image first!', 'error');
return;
}

const format = document.querySelector('input[name="format"]:checked').value;
const extension = format === 'jpeg' ? 'jpg' : format;
const fileName = originalFile.name.split('.')[0] + '_compressed.' + extension;

const url = URL.createObjectURL(compressedBlob);
const a = document.createElement('a');
a.href = url;
a.download = fileName;
a.click();

showNotification('Download started!', 'success');
}

// Reset tool
function resetTool() {
originalFile = null;
originalImage = null;
compressedImageUrl = null;
compressedBlob = null;

fileInput.value = '';
fileInfo.textContent = 'No file selected';

compressionControls.style.display = 'none';
previewSection.style.display = 'none';
actionButtons.style.display = 'none';

originalContainer.innerHTML = '';
compressedContainer.innerHTML = '';

qualitySlider.value = 80;
qualityValue.textContent = '80%';

document.getElementById('formatJpeg').checked = true;

showNotification('Tool reset successfully', 'info');
}

// Format bytes
function formatBytes(bytes, decimals = 2) {
if (bytes === 0) return '0 Bytes';

const k = 1024;
const dm = decimals < 0 ? 0 : decimals;
const sizes = ['Bytes', 'KB', 'MB', 'GB'];

const i = Math.floor(Math.log(bytes) / Math.log(k));

return parseFloat((bytes / Math.pow(k, i)).toFixed(dm)) + ' ' + sizes[i];
}

// Show notification
function showNotification(message, type = 'success') {
notification.textContent = message;
notification.style.background = type === 'success' ? '#10b981' :
type === 'error' ? '#ef4444' : '#667eea';
notification.style.display = 'block';

setTimeout(() => {
notification.style.display = 'none';
}, 3000);
}
