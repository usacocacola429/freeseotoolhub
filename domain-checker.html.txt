function populateTLDGrid(category) {
const grid = document.getElementById('tldGrid');
grid.innerHTML = '';

let tldList = [];
if (category === 'all') {
tldList = [...tlds.popular, ...tlds.country, ...tlds.professional, ...tlds.newgtlds];
} else {
tldList = tlds[category] || [];
}

// Remove duplicates
tldList = [...new Set(tldList)];

tldList.forEach(tld => {
const item = document.createElement('div');
item.className = 'tld-item';
item.innerHTML = `
<input type="checkbox" value="${tld}" ${selectedTLDs.includes(tld) ? 'checked' : ''} onchange="toggleTLD('${tld}')">
<span>${tld}</span>
`;
grid.appendChild(item);
});
}

function filterTLDs(category) {
document.querySelectorAll('.category-btn').forEach(btn => {
btn.classList.remove('active');
});
event.target.classList.add('active');
populateTLDGrid(category);
}

function toggleTLD(tld) {
if (selectedTLDs.includes(tld)) {
selectedTLDs = selectedTLDs.filter(t => t !== tld);
} else {
selectedTLDs.push(tld);
}
}

function selectAllTLDs() {
document.querySelectorAll('.tld-item input').forEach(checkbox => {
checkbox.checked = true;
const tld = checkbox.value;
if (!selectedTLDs.includes(tld)) {
selectedTLDs.push(tld);
}
});
}

function deselectAllTLDs() {
document.querySelectorAll('.tld-item input').forEach(checkbox => {
checkbox.checked = false;
});
selectedTLDs = [];
}

function toggleBulk() {
const bulkMode = document.getElementById('bulkToggle').checked;

document.getElementById('bulkSearch').style.display = bulkMode ? 'block' : 'none';
}

function checkDomain() {
const domain = document.getElementById('domainName').value.trim();
const primaryTld = document.getElementById('primaryTld').value;

if (!domain) {
alert('Please enter a domain name');
return;
}

// Show loading
document.getElementById('resultsSection').style.display = 'block';
document.getElementById('resultsBody').innerHTML = '<tr><td colspan="5" style="text-align: center;">Checking availability...</td></tr>';

// Simulate API call (replace with actual API)

// Check primary TLD
results.push(checkSingleDomain(domain, primaryTld));

// Check selected TLDs

if (tld !== primaryTld) {
results.push(checkSingleDomain(domain, tld));
}
});

currentResults = results;
displayResults(results);

}, 1000);
}

function checkBulkDomains() {
const bulkText = document.getElementById('bulkDomains').value;
const domains = bulkText.split('\n').filter(d => d.trim());

if (domains.length === 0) {
alert('Please enter at least one domain');
return;
}

if (domains.length > 50) {
alert('Maximum 50 domains allowed');
return;
}

document.getElementById('resultsSection').style.display = 'block';
document.getElementById('resultsBody').innerHTML = '<tr><td colspan="5" style="text-align: center;">Checking domains...</td></tr>';

setTimeout(() => {
const results = [];
domains.forEach(domain => {
const [name, tld] = parseDomain(domain.trim());
if (name && tld) {
results.push(checkSingleDomain(name, tld));
}
});

currentResults = results;
displayResults(results);
}, 1500);
}

function parseDomain(domain) {
const parts = domain.split('.');
if (parts.length >= 2) {
const tld = '.' + parts.slice(1).join('.');
const name = parts[0];
return [name, tld];
}
return [null, null];
}

function checkSingleDomain(name, tld) {
// Simulate domain check (replace with actual API)
const random = Math.random();
let status, price;

if (random < 0.3) {
status = 'available';
price = domainPrices[tld]?.regular || '$9.99';
} else if (random < 0.5) {
status = 'premium';
price = domainPrices[tld]?.premium || '$29.99';
} else {
status = 'unavailable';
price = '-';
}

return {
domain: name,
tld: tld,
fullDomain: name + tld,
status: status,
price: price
};
}
