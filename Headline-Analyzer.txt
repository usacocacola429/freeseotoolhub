if (!headline) {
showMessage("कृपया एक हेडलाइन डालें!", "error");
return;
}

// Show loader
loader.style.display = 'block';
hideResults();

// Simulate analysis (in real app, this would be server-side)

displayResults(analysis);

function performAnalysis(headline, keyword, type) {
const charCount = headline.length;
const wordCount = headline.split(/\s+/).length;

// Calculate length score
let lengthScore = 0;
let lengthFeedback = "";
let lengthStatus = "good";

if (charCount < 40) {
lengthScore = 4;
lengthFeedback = "हेडलाइन बहुत छोटी है। 50-60 अक्षर आदर्श हैं।";
lengthStatus = "bad";
} else if (charCount >= 40 && charCount <= 60) {
lengthScore = 10;
lengthFeedback = "हेडलाइन लंबाई आदर्श है! (Google 50-60 अक्षर दिखाता है)";
lengthStatus = "good";
} else if (charCount > 60 && charCount <= 70) {
lengthScore = 7;
lengthFeedback = "हेडलाइन थोड़ी लंबी है। कुछ अक्षर कम करने की कोशिश करें।";
lengthStatus = "warning";
} else {
lengthScore = 3;
lengthFeedback = "हेडलाइन बहुत लंबी है। Google इसे काट देगा।";
lengthStatus = "bad";
}

// Keyword analysis
let keywordScore = 0;
let keywordFeedback = "";
let keywordStatus = "good";

if (keyword) {
const keywordLower = keyword.toLowerCase();
const headlineLower = headline.toLowerCase();

if (headlineLower.includes(keywordLower)) {
// Check position
if (headlineLower.indexOf(keywordLower) < 30) {
keywordScore = 10;
keywordFeedback = "कीवर्ड हेडलाइन की शुरुआत में है - बेहतरीन!";
keywordStatus = "good";
} else {
keywordScore = 7;
keywordFeedback = "कीवर्ड हेडलाइन में है, लेकिन शुरुआत में होना बेहतर है।";
keywordStatus = "warning";
}
} else {
keywordScore = 3;
keywordFeedback = "कीवर्ड हेडलाइन में नहीं है। कीवर्ड जोड़ने की कोशिश करें।";
keywordStatus = "bad";
}
} else {
keywordScore = 6;
keywordFeedback = "कीवर्ड निर्दिष्ट नहीं किया गया। कीवर्ड जोड़कर स्कोर बेहतर करें।";
keywordStatus = "warning";
}

// Power words analysis
let powerWordsCount = 0;
const headlineWords = headline.toLowerCase().split(/\s+/);

headlineWords.forEach(word => {
if (powerWords.includes(word) || powerWords.some(pw => word.includes(pw))) {
powerWordsCount++;
}
});

let powerScore = 0;
let powerFeedback = "";
let powerStatus = "good";

if (powerWordsCount >= 2) {
powerScore = 10;
powerFeedback = `हेडलाइन में ${powerWordsCount} पावर वर्ड्स हैं - बढ़िया!`;
powerStatus = "good";
} else if (powerWordsCount === 1) {
powerScore = 7;
powerFeedback = "हेडलाइन में 1 पावर वर्ड है। एक और जोड़ने की कोशिश करें।";
powerStatus = "warning";
} else {
powerScore = 4;
powerFeedback = "हेडलाइन में कोई पावर वर्ड नहीं है। क्लिक-थ्रू रेट बढ़ाने के लिए पावर वर्ड्स जोड़ें।";
powerStatus = "bad";
}

// Emotional words analysis
let emotionalWordsCount = 0;
headlineWords.forEach(word => {
if (emotionalWords.includes(word) || emotionalWords.some(ew => word.includes(ew))) {
emotionalWordsCount++;
}
});

let emotionalScore = 0;
let emotionalFeedback = "";
let emotionalStatus = "good";
let emotionalLevel = "";

if (emotionalWordsCount >= 2) {
emotionalScore = 10;
emotionalFeedback = `हेडलाइन में ${emotionalWordsCount} भावनात्मक शब्द हैं - उत्कृष्ट!`;
emotionalStatus = "good";
emotionalLevel = "उच्च";
} else if (emotionalWordsCount === 1) {
emotionalScore = 7;
emotionalFeedback = "हेडलाइन में 1 भावनात्मक शब्द है।";
emotionalStatus = "warning";
emotionalLevel = "मध्यम";
} else {
emotionalScore = 5;
emotionalFeedback = "हेडलाइन में भावनात्मक शब्द नहीं हैं। भावनात्मक अपील बढ़ाएं।";
emotionalStatus = "bad";
emotionalLevel = "कम";
}

// Readability analysis
const avgWordLength = charCount / wordCount;
const hasQuestion = headline.includes('?');
const hasNumber = /\d/.test(headline);

let readabilityScore = 0;
let readabilityFeedback = "";
let readabilityStatus = "good";

if (avgWordLength < 6 && wordCount <= 12) {
readabilityScore = 10;
readabilityFeedback = "हेडलाइन आसानी से पढ़ी जा सकती है - बेहतरीन!";
readabilityStatus = "good";
} else if (avgWordLength < 7 && wordCount <= 15) {
readabilityScore = 7;
readabilityFeedback = "हेडलाइन पठनीय है, लेकिन थोड़ी और सरल बना सकते हैं।";
readabilityStatus = "warning";
} else {
readabilityScore = 5;
readabilityFeedback = "हेडलाइन जटिल लगती है। सरल शब्दों का प्रयोग करें।";
readabilityStatus = "bad";
}

if (hasQuestion) {
readabilityScore += 2;
readabilityFeedback += " प्रश्नवाचक हेडलाइन जिज्ञासा बढ़ाती है।";
}

if (hasNumber) {
readabilityScore += 1;
readabilityFeedback += " संख्या होने से हेडलाइन अधिक विश्वसनीय लगती है।";
}

// Calculate total score
const totalScore = Math.round(
(lengthScore + keywordScore + powerScore + emotionalScore + readabilityScore) / 5
);

// Determine grade
let grade = "";
let gradeColor = "";
let description = "";

if (totalScore >= 90) {
grade = "A+ (उत्कृष्ट)";
gradeColor = "#10b981";
description = "आपकी हेडलाइन बेहतरीन है! इसमें सभी SEO तत्व मौजूद हैं।";
} else if (totalScore >= 80) {
grade = "A (बहुत अच्छा)";
gradeColor = "#10b981";
description = "आपकी हेडलाइन अच्छी है, लेकिन कुछ छोटे सुधार संभव हैं।";
} else if (totalScore >= 70) {
grade = "B (अच्छा)";
gradeColor = "#f59e0b";
description = "आपकी हेडलाइन ठीक है, लेकिन सुधार की गुंजाइश है।";
} else if (totalScore >= 60) {
grade = "C (औसत)";
gradeColor = "#f59e0b";
description = "हेडलाइन में सुधार की आवश्यकता है। नीचे दिए सुझावों का पालन करें।";
} else {
grade = "D (सुधार आवश्यक)";
gradeColor = "#ef4444";
description = "हेडलाइन में महत्वपूर्ण सुधार की आवश्यकता है।";
}

// Generate suggestions
const suggestions = generateSuggestions(
headline, keyword, type,
{ charCount, wordCount, lengthScore, keywordScore, powerScore, emotionalScore, readabilityScore }
);

// Generate SERP preview
const serpTitleText = headline.length > 60 ? headline.substring(0, 57) + "..." : headline;
const domain = "freeseotoolhub.com";
const path = type === "blog" ? "blog" : type === "article" ? "articles" : "resources";
const serpUrlText = `https://www.${domain} › ${path} › ${headline.substring(0, 20).toLowerCase().replace(/\s+/g, '-')}`;
const serpDescText = generateSERPDescription(headline, type);

return {
headline,
keyword,
type,
scores: {
total: totalScore,
length: lengthScore,
keyword: keywordScore,
power: powerScore,
emotional: emotionalScore,
readability: readabilityScore
},
feedback: {
length: { text: lengthFeedback, status: lengthStatus },
keyword: { text: keywordFeedback, status: keywordStatus },
power: { text: powerFeedback, status: powerStatus },
emotional: { text: emotionalFeedback, status: emotionalStatus, level: emotionalLevel },
readability: { text: readabilityFeedback, status: readabilityStatus }
},
counts: {
characters: charCount,
words: wordCount,
powerWords: powerWordsCount,
emotionalWords: emotionalWordsCount
},
grade: {
text: grade,
color: gradeColor,
description: description
},
suggestions: suggestions,
serp: {
title: serpTitleText,
url: serpUrlText,
description: serpDescText
}
};
}

function generateSuggestions(headline, keyword, type, metrics) {
const suggestions = [];

// Length suggestion
if (metrics.charCount < 40) {
suggestions.push("हेडलाइन लंबी करें। आदर्श लंबाई 50-60 अक्षर है।");
} else if (metrics.charCount > 70) {
suggestions.push("हेडलाइन छोटी करें। 60 अक्षर से कम रखने का प्रयास करें।");
}

// Keyword suggestion
if (keyword && !headline.toLowerCase().includes(keyword.toLowerCase())) {
suggestions.push(`कीवर्ड "${keyword}" हेडलाइन में जोड़ें। शुरुआत में रखने का प्रयास करें।`);
} else if (keyword && headline.toLowerCase().indexOf(keyword.toLowerCase()) > 30) {
suggestions.push(`कीवर्ड "${keyword}" हेडलाइन की शुरुआत में ले जाएँ।`);
}

// Power words suggestion
if (metrics.powerWords < 2) {
suggestions.push("क्लिक-थ्रू रेट बढ़ाने के लिए 1-2 पावर वर्ड्स जोड़ें। जैसे: 'शीर्ष', 'सर्वोत्तम', 'मुफ्त', 'शक्तिशाली'");
}

// Emotional words suggestion
if (metrics.emotionalWords < 1) {
suggestions.push("भावनात्मक शब्द जोड़कर पाठकों से जुड़ें। जैसे: 'आश्चर्यजनक', 'प्रेरणादायक', 'आसान'");
}

// Type-specific suggestions
if (type === "blog") {
suggestions.push("ब्लॉग हेडलाइन में प्रश्न या संख्या का उपयोग करने से जिज्ञासा बढ़ती है।");
} else if (type === "product") {
suggestions.push("प्रोडक्ट हेडलाइन में लाभ और विशेष विशेषताओं पर ध्यान दें।");
} else if (type === "video") {
suggestions.push("यूट्यूब हेडलाइन में ब्रैकेट में कीवर्ड जोड़ें, जैसे: [2024] या [हिंदी में]");
}

// Readability suggestion
if (metrics.words > 15) {
suggestions.push("हेडलाइन को सरल और संक्षिप्त रखें। 10-12 शब्द आदर्श हैं।");
}

// Add at least 3 suggestions
if (suggestions.length < 3) {
suggestions.push("हेडलाइन में संख्या जोड़ने से क्लिक-थ्रू रेट 15% तक बढ़ सकता है।");
suggestions.push("नकारात्मक शब्दों से बचें। सकारात्मक और समाधान-उन्मुख भाषा का प्रयोग करें।");
suggestions.push("SERP में बेहतर दिखने के लिए हेडलाइन की शुरुआत में सबसे महत्वपूर्ण जानकारी रखें।");
}

return suggestions;
}

function generateSERPDescription(headline, type) {
const descriptions = {
blog: `इस ब्लॉग पोस्ट में हम ${headline.split(' ').slice(0, 3).join(' ')} के बारे में विस्तार से चर्चा करेंगे। स्टेप बाई स्टेप गाइड, उपयोगी टिप्स और प्रैक्टिकल उदाहरण जानें।`,
article: `यह आर्टिकल ${headline} पर एक संपूर्ण मार्गदर्शिका है। विशेषज्ञ सलाह, शोध आंकड़े और व्यावहारिक अनुप्रयोग यहाँ पाएं।`,
product: `हमारा ${headline} सर्वोत्तम गुणवत्ता और कीमत में उपलब्ध है। विशेष विशेषताएं, तकनीकी विवरण और ग्राहक समीक्षाएं देखें।`,
video: `इस वीडियो में हम ${headline} के बारे में बात करेंगे। पूरा ट्यूटोरियल, डेमो और व्यावहारिक उपयोग जानने के लिए देखें।`,
social: `${headline} - पूरी जानकारी, नवीनतम अपडेट और विशेष ऑफ़र पाने के लिए हमारे सोशल मीडिया पेज को फॉलो करें।`
};

return descriptions[type] || `यह पेज ${headline} के बारे में विस्तृत जानकारी प्रदान करता है। संपूर्ण मार्गदर्शिका, उपयोगी संसाधन और नवीनतम अपडेट यहाँ पाएं।`;
}

function displayResults(analysis) {
// Show all result sections
scoreSection.style.display = 'block';
analysisResults.style.display = 'block';
suggestionsBox.style.display = 'block';
serpPreview.style.display = 'block';
resultActions.style.display = 'flex';

// Update score
scoreValue.textContent = analysis.scores.total;
scoreGrade.textContent = analysis.grade.text;
scoreGrade.style.color = analysis.grade.color;
scoreDescription.textContent = analysis.grade.description;

// Update score circle color based on score
if (analysis.scores.total >= 80) {
scoreCircle.style.background = "linear-gradient(135deg, #10b981, #059669)";
} else if (analysis.scores.total >= 60) {
scoreCircle.style.background = "linear-gradient(135deg, #f59e0b, #d97706)";
} else {
scoreCircle.style.background = "linear-gradient(135deg, #ef4444, #dc2626)";
}

// Update length analysis
lengthDesc.textContent = analysis.feedback.length.text;
lengthValue.textContent = `${analysis.counts.characters} अक्षर`;
updateIcon(lengthIcon, analysis.feedback.length.status);

// Update keyword analysis
keywordDesc.textContent = analysis.feedback.keyword.text;
keywordValue.textContent = analysis.keyword ? "मौजूद" : "नहीं";
updateIcon(keywordIcon, analysis.feedback.keyword.status);

// Update power words analysis
powerDesc.textContent = analysis.feedback.power.text;
powerValue.textContent = `${analysis.counts.powerWords}`;
updateIcon(powerIcon, analysis.feedback.power.status);

// Update emotional analysis
emotionalDesc.textContent = analysis.feedback.emotional.text;
emotionalValue.textContent = analysis.feedback.emotional.level;
updateIcon(emotionalIcon, analysis.feedback.emotional.status);

// Update readability analysis
readabilityDesc.textContent = analysis.feedback.readability.text;
readabilityValue.textContent = analysis.counts.words > 15 ? "मध्यम" : "अच्छा";
updateIcon(readabilityIcon, analysis.feedback.readability.status);

// Update suggestions
suggestionsList.innerHTML = "";
analysis.suggestions.forEach((suggestion, index) => {
const suggestionItem = document.createElement('div');
suggestionItem.className = 'suggestion-item';
suggestionItem.innerHTML = `<strong>${index + 1}.</strong> ${suggestion}`;
suggestionsList.appendChild(suggestionItem);
});

// Update SERP preview
serpTitle.textContent = analysis.serp.title;
serpUrl.textContent = analysis.serp.url;
serpDesc.textContent = analysis.serp.description;
}

function updateIcon(iconElement, status) {
iconElement.className = "analysis-icon";

if (status === "good") {
iconElement.classList.add("analysis-good");
iconElement.innerHTML = '<i class="fas fa-check-circle"></i>';
} else if (status === "warning") {
iconElement.classList.add("analysis-warning");
iconElement.innerHTML = '<i class="fas fa-exclamation-triangle"></i>';
} else {
iconElement.classList.add("analysis-bad");
iconElement.innerHTML = '<i class="fas fa-times-circle"></i>';
}
}

function copyHeadline() {
const headline = headlineInput.value.trim();

if (!headline) {
showMessage("कॉपी करने के लिए कोई हेडलाइन नहीं!", "error");
return;
}

navigator.clipboard.writeText(headline)
.then(() => {
showMessage("हेडलाइन कॉपी हो गई!", "success");
})
.catch(err => {
showMessage("कॉपी करने में त्रुटि: " + err, "error");
});
}

function shareResults() {
const headline = headlineInput.value.trim();
const score = scoreValue.textContent;
const grade = scoreGrade.textContent;

if (!headline || !score || score === "0") {
showMessage("कृपया पहले हेडलाइन का विश्लेषण करें!", "error");
return;
}

const shareText = `मेरी हेडलाइन "${headline}" ने ${score}/100 स्कोर किया! ग्रेड: ${grade}. Free SEO Tool Hub - Headline Analyzer का उपयोग करके अपनी हेडलाइन का विश्लेषण करें: ${window.location.href}`;

if (navigator.share) {
navigator.share({
title: 'हेडलाइन एनालाइज़र रिजल्ट',
text: shareText,
url: window.location.href
})
.then(() => showMessage("रिजल्ट शेयर किया गया!", "success"))
.catch(err => showMessage("शेयर करने में त्रुटि: " + err, "error"));
} else {
// Fallback: Copy to clipboard
navigator.clipboard.writeText(shareText)
.then(() => {
showMessage("रिजल्ट कॉपी हो गया! आप इसे शेयर कर सकते हैं।", "success");
})
.catch(err => {
showMessage("शेयर करने में त्रुटि: " + err, "error");
});
}

successMessage.style.background = "#f59e0b";
} else {
successMessage.style.background = "#10B981";
}

successMessage.style.display = 'flex';

setTimeout(() => {
successMessage.style.display = 'none';
}, 3000);
}
