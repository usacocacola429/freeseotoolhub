function extractHeadingsFromHTML(html) {
// Parse HTML string to extract headings
const headings = [];

// Create a temporary div to parse HTML
const tempDiv = document.createElement('div');
tempDiv.innerHTML = html;

// Find all heading elements
const headingElements = tempDiv.querySelectorAll('h1, h2, h3, h4, h5, h6');

headingElements.forEach((element, index) => {
headings.push({
tag: element.tagName.toUpperCase(),
text: element.textContent.trim(),
count: index + 1
});
});

// If no headings found in HTML, check for heading-like patterns
if (headings.length === 0) {
const regex = /<(h[1-6])(?:\s[^>]*)?>(.*?)<\/\1>/gi;
let match;
let count = 1;

while ((match = regex.exec(html)) !== null) {
headings.push({
tag: match[1].toUpperCase(),
text: match[2].replace(/<[^>]*>/g, '').trim(),
count: count++
});
}
}

return headings;
}

function extractHeadingsFromManual(text) {
const headings = [];
const lines = text.split('\n');
let count = 1;

lines.forEach(line => {
line = line.trim();
if (!line) return;

// Check for patterns like "H1: Title" or "# Title" or "<h1>Title</h1>"
const h1Match = line.match(/^(H1|h1|#):\s*(.+)$/i) || line.match(/<h1(?:\s[^>]*)?>(.*?)<\/h1>/i);
const h2Match = line.match(/^(H2|h2|##):\s*(.+)$/i) || line.match(/<h2(?:\s[^>]*)?>(.*?)<\/h2>/i);
const h3Match = line.match(/^(H3|h3|###):\s*(.+)$/i) || line.match(/<h3(?:\s[^>]*)?>(.*?)<\/h3>/i);
const h4Match = line.match(/^(H4|h4|####):\s*(.+)$/i) || line.match(/<h4(?:\s[^>]*)?>(.*?)<\/h4>/i);
const h5Match = line.match(/^(H5|h5|#####):\s*(.+)$/i) || line.match(/<h5(?:\s[^>]*)?>(.*?)<\/h5>/i);
const h6Match = line.match(/^(H6|h6|######):\s*(.+)$/i) || line.match(/<h6(?:\s[^>]*)?>(.*?)<\/h6>/i);

if (h1Match) {
headings.push({ tag: 'H1', text: h1Match[2] || h1Match[1], count: count++ });
} else if (h2Match) {
headings.push({ tag: 'H2', text: h2Match[2] || h2Match[1], count: count++ });
} else if (h3Match) {
headings.push({ tag: 'H3', text: h3Match[2] || h3Match[1], count: count++ });
} else if (h4Match) {
headings.push({ tag: 'H4', text: h4Match[2] || h4Match[1], count: count++ });
} else if (h5Match) {
headings.push({ tag: 'H5', text: h5Match[2] || h5Match[1], count: count++ });
} else if (h6Match) {
headings.push({ tag: 'H6', text: h6Match[2] || h6Match[1], count: count++ });
} else {
// If no pattern matched, assume it's a heading text
headings.push({ tag: 'H2', text: line, count: count++ });
}
});

return headings;
}

function updateStatistics(headings) {
// Count headings by type
const counts = {
total: headings.length,
h1: headings.filter(h => h.tag === 'H1').length,
h2: headings.filter(h => h.tag === 'H2').length,
h3: headings.filter(h => h.tag === 'H3').length,
h4: headings.filter(h => h.tag === 'H4').length,
h5: headings.filter(h => h.tag === 'H5').length,
h6: headings.filter(h => h.tag === 'H6').length
};

// Update UI
document.getElementById('totalHeadingsCount').textContent = counts.total;
document.getElementById('h1Count').textContent = counts.h1;
document.getElementById('h2Count').textContent = counts.h2;

// Calculate SEO score (simplified)
let score = 100;

// Deductions for issues
if (counts.h1 === 0) score -= 40; // No H1 is critical
else if (counts.h1 > 1) score -= (counts.h1 - 1) * 20; // Multiple H1s

if (counts.total === 0) score = 0; // No headings

// Adjust based on hierarchy
if (counts.h1 > 0 && counts.h2 === 0) score -= 20; // H1 but no H2s
if (counts.h2 > 0 && counts.h1 === 0) score -= 30; // H2s but no H1

// Ensure score is between 0 and 100
score = Math.max(0, Math.min(100, score));

// Update score with color coding
const seoScoreElement = document.getElementById('seoScore');
seoScoreElement.textContent = `${Math.round(score)}%`;

if (score >= 80) {
seoScoreElement.className = 'stat-number text-success';
} else if (score >= 50) {
seoScoreElement.className = 'stat-number text-warning';
} else {
seoScoreElement.className = 'stat-number text-danger';
}
}

text: '❌ No H1 heading found. Every page should have exactly one H1 heading.'
});
} else if (counts.h1 > 1) {
recommendations.push({
type: 'error',
text: `❌ Multiple H1 headings found (${counts.h1}). Use only one H1 per page.`
});
} else {
recommendations.push({
type: 'success',
text: '✅ Perfect! You have exactly one H1 heading.'
});
}

if (counts.h2 === 0 && counts.total > 1) {
recommendations.push({
type: 'warning',
text: '⚠️ No H2 headings found. Consider using H2 headings to organize content sections.'
});
} else if (counts.h2 > 0) {
recommendations.push({
type: 'success',
text: `✅ Good! You have ${counts.h2} H2 heading(s) for content organization.`
});
}

if (counts.total === 0) {
recommendations.push({
type: 'error',
text: '❌ No headings found. Headings are essential for SEO and user experience.'
});
}

// Check hierarchy
let hasHierarchyIssue = false;
let lastLevel = 0;

for (let heading of headings) {
const level = parseInt(heading.tag.charAt(1));
if (level > lastLevel + 1) {
hasHierarchyIssue = true;
break;
}
lastLevel = level;
}

if (hasHierarchyIssue) {
recommendations.push({
type: 'warning',
text: '⚠️ Heading hierarchy may have gaps. Try to follow a logical order: H1 → H2 → H3 etc.'
});
} else {
recommendations.push({
type: 'success',
text: '✅ Heading hierarchy appears to be logical and well-structured.'
});
}

// Check for empty headings
const emptyHeadings = headings.filter(h => !h.text || h.text.trim() === '');
if (emptyHeadings.length > 0) {
recommendations.push({
type: 'error',
text: `❌ Found ${emptyHeadings.length} empty heading(s). All headings should have descriptive text.`
});
}

// Display recommendations
let html = '';
recommendations.forEach(rec => {
const alertClass = rec.type === 'error' ? 'alert-danger' :
rec.type === 'warning' ? 'alert-warning' : 'alert-success';

html += `
<div class="alert ${alertClass} mb-2">
${rec.text}
</div>`;
});

// Add general tips
html += `
<div class="mt-3">
<h6>Additional Tips:</h6>
<ul class="mb-0">
<li>Include primary keywords in H1 and secondary keywords in H2/H3 headings</li>
<li>Keep headings concise (under 70 characters for H1, under 60 for others)</li>
<li>Use headings to create a logical content outline</li>
<li>Ensure headings accurately describe the content that follows</li>
</ul>
</div>`;

seoRecommendations.innerHTML = html;
}

// Create report content
const url = urlInput.value || 'Manual Input';
const headingsList = document.querySelectorAll('.heading-item');

let report = `H1-H6 Heading Structure Report\n`;
report += `================================\n`;
report += `URL: ${url}\n`;
report += `Date: ${new Date().toLocaleDateString()}\n`;
report += `Time: ${new Date().toLocaleTimeString()}\n`;
report += `Total Headings: ${document.getElementById('totalHeadingsCount').textContent}\n\n`;

report += `HEADING BREAKDOWN:\n`;
report += `H1: ${document.getElementById('h1Count').textContent}\n`;
report += `H2: ${document.getElementById('h2Count').textContent}\n`;
report += `SEO Score: ${document.getElementById('seoScore').textContent}\n\n`;

report += `DETAILED HEADING LIST:\n`;
report += `=====================\n`;

headingsList.forEach((item, index) => {
const tag = item.querySelector('.heading-tag').textContent;
const text = item.querySelector('strong').textContent;
report += `${index + 1}. ${tag}: ${text}\n`;
});

report += `\nSEO RECOMMENDATIONS:\n`;
report += `===================\n`;

const recommendations = document.querySelectorAll('#seoRecommendations .alert');
recommendations.forEach(rec => {
report += `- ${rec.textContent.trim()}\n`;
});

// Create blob and download
const blob = new Blob([report], { type: 'text/plain' });
const urlBlob = URL.createObjectURL(blob);
const a = document.createElement('a');
a.href = urlBlob;
a.download = `heading-report-${new Date().toISOString().slice(0,10)}.txt`;
document.body.appendChild(a);
a.click();
document.body.removeChild(a);
URL.revokeObjectURL(urlBlob);
}
