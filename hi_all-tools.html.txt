document.addEventListener('DOMContentLoaded', function() {
const toolCards = document.querySelectorAll('.tool-card');
const searchInput = document.getElementById('toolSearch');
const noResults = document.getElementById('noResults');
const filterBtns = document.querySelectorAll('.filter-btn');
const categories = document.querySelectorAll('.category-header');
const categoryDescs = document.querySelectorAll('.category-desc');

// Update dynamic counts
const globalCards = document.querySelectorAll('.tool-card.global').length;
const proCards = document.querySelectorAll('.tool-card.pro').length;

document.getElementById('totalToolCount').textContent = toolCards.length;
document.getElementById('globalToolCount').textContent = globalCards;
document.getElementById('proToolCount').textContent = proCards;

// Language preference saver
const langLinks = document.querySelectorAll('.dropdown-item');
langLinks.forEach(link => {
link.addEventListener('click', function(e) {
const href = this.getAttribute('href');
const lang = href.split('/')[1];
localStorage.setItem('preferredLanguage', lang);
window.location.href = href;
});
});

function filterTools(searchTerm) {
let visible = 0;
toolCards.forEach(card => {
const title = card.querySelector('h3')?.textContent.toLowerCase() || '';
const desc = card.querySelector('p')?.textContent.toLowerCase() || '';
const tags = card.querySelector('.tool-tags')?.textContent.toLowerCase() || '';
const match = title.includes(searchTerm) || desc.includes(searchTerm) || tags.includes(searchTerm);
card.closest('.col-md-4, .col-lg-3').style.display = match ? 'block' : 'none';
if(match) visible++;
});

categories.forEach((cat, index) => {
let anyVisible = false;
let next = cat.nextElementSibling;
if(next && next.classList.contains('category-desc')) next = next.nextElementSibling;
while(next && next.classList.contains('row')) {
if(next.querySelector('.col-md-4[style*="block"], .col-lg-3[style*="block"]')) anyVisible = true;
next = next.nextElementSibling;
}
cat.style.display = anyVisible ? 'flex' : 'none';
if(categoryDescs[index]) categoryDescs[index].style.display = anyVisible ? 'flex' : 'none';
});

noResults.style.display = (visible === 0 && searchTerm !== '') ? 'block' : 'none';
}

searchInput.addEventListener('input', () => filterTools(searchInput.value.toLowerCase().trim()));

filterBtns.forEach(btn => {
btn.addEventListener('click', function() {
filterBtns.forEach(b => b.classList.remove('active'));
this.classList.add('active');
const filter = this.dataset.filter;

toolCards.forEach(card => {
const col = card.closest('.col-md-4, .col-lg-3');
if(filter === 'all') col.style.display = 'block';
else if(filter === 'global') col.style.display = card.classList.contains('global') ? 'block' : 'none';
else if(filter === 'pro') col.style.display = card.classList.contains('pro') ? 'block' : 'none';
else if(filter === 'hot') {
const isHot = card.classList.contains('hot') || card.querySelector('.status-hot');
col.style.display = isHot ? 'block' : 'none';
}
});

categories.forEach((cat, index) => {
let anyVisible = false;
let next = cat.nextElementSibling;
if(next && next.classList.contains('category-desc')) next = next.nextElementSibling;
while(next && next.classList.contains('row')) {
if(next.querySelector('.col-md-4[style*="block"], .col-lg-3[style*="block"]')) anyVisible = true;
next = next.nextElementSibling;
}
cat.style.display = anyVisible ? 'flex' : 'none';
if(categoryDescs[index]) categoryDescs[index].style.display = anyVisible ? 'flex' : 'none';
});
});
});

// Clickable cards
toolCards.forEach(card => {
card.addEventListener('click', function(e) {
if(e.target.tagName !== 'A' && !e.target.closest('a') && !e.target.closest('button')) {
const link = this.querySelector('a');
if(link) window.location = link.href;
}
});
});
});
