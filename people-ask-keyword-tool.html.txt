<script>
function generateQuestions() {
const topic = document.getElementById('topicInput').value.trim();
const loader = document.querySelector('.loader');
const resultsArea = document.getElementById('resultsArea');
const list = document.getElementById('questionsList');

if(!topic) {
alert("Please enter a topic.");
return;
}

// Show Loader
loader.style.display = 'inline-block';
resultsArea.style.display = 'none';
list.innerHTML = '';

// Simulate API Call
setTimeout(() => {
loader.style.display = 'none';
resultsArea.style.display = 'block';

// Mock Data Generation Logic
const questions = [
`What is ${topic}?`,
`How does ${topic} work?`,
`Why is ${topic} important?`,
`What are the benefits of ${topic}?`,
`Is ${topic} safe?`,
`How to start with ${topic}?`,
`What tools are needed for ${topic}?`,
`What is the best way to learn ${topic}?`,
`Can I make money with ${topic}?`,
`What are the common mistakes in ${topic}?`,
`How much does ${topic} cost?`,
`What is the future of ${topic}?`,
`Who can benefit from ${topic}?`,
`Is ${topic} right for beginners?`,
`What certifications are needed for ${topic}?`,
`How long does it take to master ${topic}?`,
`What are the challenges in ${topic}?`,
`How to improve skills in ${topic}?`,
`Where can I find resources for ${topic}?`,
`What is the difference between ${topic} and similar topics?`
];

questions.forEach(q => {
const li = document.createElement('div');
li.className = 'list-group-item question-item';
li.textContent = q;
list.appendChild(li);
});

}, 1000);
}

function copyAllQuestions() {
const items = document.querySelectorAll('.question-item');
let text = "";
items.forEach(item => {
text += item.textContent + "\n";
});
navigator.clipboard.writeText(text).then(() => {
alert("All questions copied!");
});
}
