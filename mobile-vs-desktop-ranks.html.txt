<script>
document.addEventListener('DOMContentLoaded', function() {
const analyzeBtn = document.getElementById('analyzeBtn');
const loading = document.getElementById('loading');
const resultsSection = document.getElementById('resultsSection');
const tableBody = document.getElementById('tableBody');
const statsCards = document.getElementById('statsCards');
const insightsText = document.getElementById('insightsText');
const mobileTips = document.getElementById('mobileTips');
const desktopTips = document.getElementById('desktopTips');

let rankingChart, comparisonChart;

// Sample keywords data
const keywords = [
{ keyword: 'seo tools', volume: 12500, mobile: 3, desktop: 5 },
{ keyword: 'keyword research tool', volume: 8200, mobile: 7, desktop: 4 },
{ keyword: 'backlink checker', volume: 6300, mobile: 12, desktop: 9 },
{ keyword: 'rank tracker', volume: 5400, mobile: 8, desktop: 11 },
{ keyword: 'site audit tool', volume: 4300, mobile: 15, desktop: 14 },
{ keyword: 'competitor analysis', volume: 3900, mobile: 22, desktop: 18 },
{ keyword: 'local seo tool', volume: 3500, mobile: 5, desktop: 8 },
{ keyword: 'technical seo', volume: 3100, mobile: 19, desktop: 21 },
{ keyword: 'content optimizer', volume: 2800, mobile: 11, desktop: 7 },
{ keyword: 'seo audit', volume: 2600, mobile: 6, desktop: 6 },
{ keyword: 'google rankings', volume: 2400, mobile: 9, desktop: 12 },
{ keyword: 'seo software', volume: 2200, mobile: 14, desktop: 10 }
];

function generateRankings() {
return keywords.map(k => ({
...k,
mobile: k.mobile + Math.floor(Math.random() * 3) - 1,
desktop: k.desktop + Math.floor(Math.random() * 3) - 1
}));
}

function updateStats(rankings) {
const avgMobile = Math.round(rankings.reduce((sum, k) => sum + k.mobile, 0) / rankings.length);
const avgDesktop = Math.round(rankings.reduce((sum, k) => sum + k.desktop, 0) / rankings.length);
const betterMobile = rankings.filter(k => k.mobile < k.desktop).length;
const betterDesktop = rankings.filter(k => k.desktop < k.mobile).length;
const same = rankings.length - betterMobile - betterDesktop;

statsCards.innerHTML = `
<div class="stat-card">
<i class="fas fa-mobile-alt"></i>
<div class="stat-label">Avg Mobile Rank</div>
<div class="stat-value">${avgMobile}</div>
<div class="stat-sub">Top 3: ${rankings.filter(k => k.mobile <= 3).length}</div>
</div>
<div class="stat-card">
<i class="fas fa-desktop"></i>
<div class="stat-label">Avg Desktop Rank</div>
<div class="stat-value">${avgDesktop}</div>
<div class="stat-sub">Top 3: ${rankings.filter(k => k.desktop <= 3).length}</div>
</div>
<div class="stat-card">
<i class="fas fa-arrow-up" style="color: var(--success);"></i>
<div class="stat-label">Mobile Better</div>
<div class="stat-value">${betterMobile}</div>
<div class="stat-sub">keywords</div>
</div>
<div class="stat-card">
<i class="fas fa-arrow-down" style="color: var(--danger);"></i>
<div class="stat-label">Desktop Better</div>
<div class="stat-value">${betterDesktop}</div>
<div class="stat-sub">keywords</div>
</div>
`;

// Insights
insightsText.innerHTML = `
<p><i class="fas fa-circle" style="color: var(--primary);"></i> <strong>Mobile vs Desktop:</strong> Your mobile rankings are ${avgMobile < avgDesktop ? 'better' : 'worse'} than desktop by ${Math.abs(avgMobile - avgDesktop)} positions on average.</p>
<p><i class="fas fa-circle" style="color: var(--success);"></i> <strong>Opportunity:</strong> ${betterDesktop} keywords need mobile optimization to catch up with desktop rankings.</p>
<p><i class="fas fa-circle" style="color: var(--warning);"></i> <strong>Top Priority:</strong> Focus on "${rankings.find(k => Math.abs(k.mobile - k.desktop) > 5)?.keyword}" which has the biggest device gap.</p>
`;

// Tips
mobileTips.innerHTML = `
<ul class="list-unstyled">
<li><i class="fas fa-check text-success"></i> Improve page speed (target LCP < 2.5s)</li>
<li><i class="fas fa-check text-success"></i> Optimize for touch (buttons size > 48px)</li>
<li><i class="fas fa-check text-success"></i> Use responsive design</li>
<li><i class="fas fa-check text-success"></i> Avoid intrusive popups</li>
</ul>
`;
desktopTips.innerHTML = `
<ul class="list-unstyled">
<li><i class="fas fa-check text-primary"></i> Enhance internal linking structure</li>
<li><i class="fas fa-check text-primary"></i> Create in-depth content (1500+ words)</li>
<li><i class="fas fa-check text-primary"></i> Build quality backlinks</li>
<li><i class="fas fa-check text-primary"></i> Optimize for featured snippets</li>
</ul>
`;
}

function updateTable(rankings) {
let html = '';
rankings.forEach(k => {
const diff = k.mobile - k.desktop;
const diffClass = diff < 0 ? 'rank-better' : (diff > 0 ? 'rank-worse' : 'rank-same');
const diffIcon = diff < 0 ? '↓ Better' : (diff > 0 ? '↑ Worse' : '= Same');
const devicePref = k.mobile < k.desktop ? 'Mobile' : (k.desktop < k.mobile ? 'Desktop' : 'Equal');
const prefBadge = k.mobile < k.desktop ? 'badge-mobile' : (k.desktop < k.mobile ? 'badge-desktop' : 'bg-secondary text-white');

html += `
<tr>
<td><strong>${k.keyword}</strong></td>
<td>${k.volume.toLocaleString()}</td>
<td><span class="badge bg-primary">#${k.mobile}</span></td>
<td><span class="badge bg-secondary">#${k.desktop}</span></td>
<td class="${diffClass}">${diffIcon} (${Math.abs(diff)})</td>
<td>${diff === 0 ? '<span class="badge bg-warning">Equal</span>' : (diff < 0 ? '<span class="badge bg-success">Mobile Better</span>' : '<span class="badge bg-danger">Desktop Better</span>')}</td>
<td><span class="device-badge ${prefBadge}">${devicePref}</span></td>
</tr>
`;
});
tableBody.innerHTML = html;
}

function updateCharts(rankings) {
// Destroy existing charts
if (rankingChart) rankingChart.destroy();
if (comparisonChart) comparisonChart.destroy();

// Ranking Distribution Chart
const ctx1 = document.getElementById('rankingChart').getContext('2d');
rankingChart = new Chart(ctx1, {
type: 'doughnut',
data: {
labels: ['Mobile Better', 'Desktop Better', 'Equal'],
datasets: [{
data: [
rankings.filter(k => k.mobile < k.desktop).length,
rankings.filter(k => k.desktop < k.mobile).length,
rankings.filter(k => k.mobile === k.desktop).length
],
backgroundColor: ['#06d6a0', '#4361ee', '#ffd166'],
borderWidth: 0
}]
},
options: {
responsive: true,
maintainAspectRatio: false,
plugins: { legend: { position: 'bottom' } }
}
});

// Comparison Chart
const ctx2 = document.getElementById('comparisonChart').getContext('2d');
comparisonChart = new Chart(ctx2, {
type: 'bar',
data: {
labels: rankings.slice(0, 6).map(k => k.keyword),
datasets: [
{
label: 'Mobile Rank',
data: rankings.slice(0, 6).map(k => k.mobile),
backgroundColor: '#06d6a0',
borderRadius: 5
},
{
label: 'Desktop Rank',
data: rankings.slice(0, 6).map(k => k.desktop),
backgroundColor: '#4361ee',
borderRadius: 5
}
]
},
options: {
responsive: true,
maintainAspectRatio: false,
scales: { y: { reverse: true, beginAtZero: false, min: 0 } }
}
});
}

analyzeBtn.addEventListener('click', function() {
const domain = document.getElementById('domainInput').value;
if (!domain) {
alert('Please enter a domain');
return;
}

loading.style.display = 'block';
resultsSection.style.display = 'none';

setTimeout(() => {
const rankings = generateRankings();
updateStats(rankings);
updateTable(rankings);
updateCharts(rankings);

loading.style.display = 'none';
resultsSection.style.display = 'block';
}, 2000);
});

// Auto-run
setTimeout(() => analyzeBtn.click(), 500);
});
