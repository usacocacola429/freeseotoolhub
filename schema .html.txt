function formatJSONForDisplay(jsonString) {
let html = '';
const jsonObj = JSON.parse(jsonString);

function parseObject(obj, indent = 0) {
const indentStr = ' '.repeat(indent);

html += `<div class="code-line">${indentStr}{</div>`;

const keys = Object.keys(obj);
keys.forEach((key, index) => {
const value = obj[key];
const isLast = index === keys.length - 1;

html += `<div class="code-line">${indentStr} <span class="code-key">"${key}"</span>: `;

if (typeof value === 'object' && value !== null) {
if (Array.isArray(value)) {
html += `[</div>`;
value.forEach((item, i) => {
if (typeof item === 'object') {
parseObject(item, indent + 2);
} else {
html += `<div class="code-line">${indentStr} <span class="code-string">"${item}"</span>${i === value.length - 1 ? '' : ','}</div>`;
}
});
html += `<div class="code-line">${indentStr} ]${isLast ? '' : ','}</div>`;
} else {
parseObject(value, indent + 1);
html += `<div class="code-line">${indentStr} }${isLast ? '' : ','}</div>`;
}
} else {
if (typeof value === 'string') {
html += `<span class="code-string">"${value}"</span>`;
} else if (typeof value === 'number') {
html += `<span class="code-number">${value}</span>`;
} else if (typeof value === 'boolean') {
html += `<span class="code-boolean">${value}</span>`;
} else {
html += `<span class="code-string">"${value}"</span>`;
}
html += `${isLast ? '' : ','}</div>`;
}
});

html += `<div class="code-line">${indentStr}}</div>`;
}

parseObject(jsonObj);
return html;
}

function generateStars(rating) {
let stars = '';
const fullStars = Math.floor(rating);
const hasHalfStar = rating % 1 >= 0.5;

for (let i = 0; i < fullStars; i++) {
stars += '<i class="fas fa-star"></i>';
}

if (hasHalfStar) {
stars += '<i class="fas fa-star-half-alt"></i>';
}

const emptyStars = 5 - fullStars - (hasHalfStar ? 1 : 0);
for (let i = 0; i < emptyStars; i++) {
stars += '<i class="far fa-star"></i>';
}

return stars;
}

function displayValidationResults(schema) {
const errors = [];
const warnings = [];
const successes = [];

// Basic validation
if (!schema.url || !isValidURL(schema.url)) {
errors.push({
title: "Invalid URL",
description: "The website URL is not valid"
});
} else {
successes.push({
title: "Valid URL",
description: "Website URL is properly formatted"
});
}

if (!schema['@context'] || schema['@context'] !== 'https://schema.org') {
errors.push({
title: "Missing Context",
description: "@context must be 'https://schema.org'"
});
} else {
successes.push({
title: "Correct Context",
description: "@context is set to 'https://schema.org'"
});
}

if (!schema['@type']) {
errors.push({
title: "Missing Type",
description: "@type is required"
});
} else {
successes.push({
title: "Type Defined",
description: `Schema type is '${schema['@type']}'`
});
}

// Type-specific validation
if (schema['@type'] === 'Article') {
if (!schema.headline) {
warnings.push({
title: "Missing Headline",
description: "Article should have a headline"
});
}
if (!schema.author) {
warnings.push({
title: "Missing Author",
description: "Article should specify an author"
});
}
}

if (schema['@type'] === 'Product') {
if (!schema.offers?.price) {
warnings.push({
title: "Missing Price",
description: "Product should have a price"
});
}
}

let validationHTML = '';

if (errors.length === 0 && warnings.length === 0) {
validationHTML = `
<div class="validation-item">
<i class="fas fa-check-circle validation-icon validation-good"></i>
<div class="validation-text">
<div class="validation-title">Perfect Schema!</div>
<div class="validation-desc">All validation checks passed successfully</div>
</div>
</div>
`;
}

errors.forEach(error => {
validationHTML += `
<div class="validation-item">
<i class="fas fa-times-circle validation-icon validation-error"></i>
<div class="validation-text">
<div class="validation-title">${error.title}</div>
<div class="validation-desc">${error.description}</div>
</div>
</div>
`;
});

warnings.forEach(warning => {
validationHTML += `
<div class="validation-item">
<i class="fas fa-exclamation-triangle validation-icon validation-warning"></i>
<div class="validation-text">
<div class="validation-title">${warning.title}</div>
<div class="validation-desc">${warning.description}</div>
</div>
</div>
`;
});

successes.forEach(success => {
validationHTML += `
<div class="validation-item">
<i class="fas fa-check validation-icon validation-good"></i>
<div class="validation-text">
<div class="validation-title">${success.title}</div>
<div class="validation-desc">${success.description}</div>
</div>
</div>
`;
});

validationResults.innerHTML = validationHTML;
}

function downloadSchema() {

showMessage('कृपया पहले schema generate करें!', 'error');
return;
}

const jsonString = JSON.stringify(generatedSchema, null, 2);
const blob = new Blob([jsonString], { type: 'application/json' });
const url = URL.createObjectURL(blob);

const a = document.createElement('a');
a.href = url;
a.download = `schema-${generatedSchema['@type'].toLowerCase()}-${Date.now()}.json`;
document.body.appendChild(a);
a.click();
document.body.removeChild(a);
URL.revokeObjectURL(url);

showMessage('Schema downloaded successfully!', 'success');
}
