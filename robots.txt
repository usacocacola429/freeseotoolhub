    // Generate robots.txt
    function generateRobots() {
      const websiteUrl = document.getElementById('websiteUrl').value;
      const sitemapUrl = document.getElementById('sitemapUrl').value;
      const additional = document.getElementById('additionalDirectives').value;
      
      let robots = `# robots.txt generated by Free SEO Tool Hub
# https://freeseotoolhub.com/robots-txt-generator.html
# Created for: ${websiteUrl}
# Date: ${new Date().toISOString().split('T')[0]}

`;
      
      // Process each rule
      const rules = document.querySelectorAll('.rule-group');
      rules.forEach(rule => {
        const userAgent = rule.querySelector('.userAgent').value;
        const crawlDelay = rule.querySelector('.crawlDelay').value;
        const disallowPaths = rule.querySelector('.disallowPaths').value;
        const allowPaths = rule.querySelector('.allowPaths').value;
        
        robots += `User-agent: ${userAgent}\n`;
        
        // Add allow rules first
        if (allowPaths.trim()) {
          allowPaths.split('\n').forEach(path => {
            if (path.trim()) {
              robots += `Allow: ${path.trim()}\n`;
            }
          });
        }
        
        // Add disallow rules
        if (disallowPaths.trim()) {
          disallowPaths.split('\n').forEach(path => {
            if (path.trim()) {
              robots += `Disallow: ${path.trim()}\n`;
            }
          });
        } else {
          robots += 'Disallow: \n';
        }
        
        // Add crawl delay if specified
        if (crawlDelay.trim()) {
          robots += `Crawl-delay: ${crawlDelay.trim()}\n`;
        }
        
        robots += '\n';
      });
      
      // Add sitemap
      if (sitemapUrl.trim()) {
        robots += `Sitemap: ${sitemapUrl}\n\n`;
      }
      
      // Add additional directives
      if (additional.trim()) {
        robots += additional + '\n\n';
      }
      
      // Add footer
      robots += `# Free SEO Tool Hub - Robots.txt Generator
# More tools: https://freeseotoolhub.com/all-tools.html
`;
      
      document.getElementById('robotsPreview').textContent = robots;
      updateFileStats(robots);
      showMessage('Robots.txt generated successfully!', 'success');
    }

    // Validate robots.txt
    function validateRobots() {
      const content = document.getElementById('robotsPreview').textContent;
      
      // Basic validation checks
      const errors = [];
      
      if (!content.includes('User-agent:')) {
        errors.push('Missing User-agent directive');
      }
      
      if (!content.includes('Sitemap:')) {
        errors.push('Missing Sitemap directive (recommended)');
      }
      
      if (content.includes('Disallow: /')) {
        errors.push('Warning: Disallowing root directory may block all crawling');
      }
      
      if (errors.length > 0) {
        showMessage('Validation: ' + errors.join(', '), 'warning');
      } else {
        showMessage('âœ“ robots.txt is valid!', 'success');
      }
    }
