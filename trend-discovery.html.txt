// Initialize
window.onload = function() {
filterTrends('all');
};

function filterTrends(category) {
document.querySelectorAll('.category-btn').forEach(btn => btn.classList.remove('active'));
event.target.classList.add('active');

const grid = document.getElementById('trendsGrid');
grid.innerHTML = '';

if (category === 'all') {
// Show all trends
Object.values(trends).flat().forEach(t => addTrendCard(t));
} else {
trends[category].forEach(t => addTrendCard(t));
}
}

function searchTrends() {
const query = document.getElementById('keywordSearch').value.trim().toLowerCase();

if (!query) {
filterTrends('all');
return;
}

const grid = document.getElementById('trendsGrid');
grid.innerHTML = '';

// Search in all trends
const allTrends = Object.values(trends).flat();
const filtered = allTrends.filter(t =>
t.title.toLowerCase().includes(query) ||
t.description.toLowerCase().includes(query) ||
t.keywords.some(k => k.toLowerCase().includes(query))
);

if (filtered.length === 0) {
grid.innerHTML = `
<div style="grid-column: 1/-1; text-align: center; padding: 50px;">
<i class="fas fa-search" style="font-size: 50px; color: #94a3b8; margin-bottom: 20px;"></i>
<h3 style="color: #1e293b;">No trends found for "${query}"</h3>
<p style="color: #64748b;">Try searching with different keywords</p>
</div>
`;
} else {
filtered.forEach(t => addTrendCard(t));
}
}
