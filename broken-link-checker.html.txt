function cleanURL(url) {
// Remove protocol and trailing slashes
let clean = url.replace(/^(https?:\/\/)?/, '');
clean = clean.replace(/\/+$/, '');
return clean.toLowerCase();
}

function getLinkData(website, maxLinks) {
// Check if website is in our database
if (linkDatabase[website]) {
return linkDatabase[website].slice(0, maxLinks);
}

// Generate realistic link data for new websites
const links = [];
const linkCount = Math.min(maxLinks, Math.floor(Math.random() * 30) + 20); // 20-50 links

const statusTypes = [
{ status: 200, type: 'working', chance: 0.7 },
{ status: 404, type: 'broken', chance: 0.2 },
{ status: 301, type: 'redirect', chance: 0.08 },
{ status: 500, type: 'broken', chance: 0.02 }
];

const pages = ['Homepage', 'About', 'Contact', 'Blog', 'Products', 'Services', 'FAQ', 'Resources'];

for (let i = 0; i < linkCount; i++) {
const random = Math.random();
let cumulative = 0;
let selectedStatus;

for (const statusType of statusTypes) {
cumulative += statusType.chance;
if (random <= cumulative) {
selectedStatus = statusType;
break;
}
}

const pageType = i % 3 === 0 ? 'internal' : (i % 3 === 1 ? 'external' : 'resource');
let url;

if (pageType === 'internal') {
const paths = ['page', 'article', 'post', 'product', 'service'];
const path = paths[Math.floor(Math.random() * paths.length)];
url = `https://${website}/${path}-${Math.floor(Math.random() * 1000)}`;
} else if (pageType === 'external') {
const domains = ['external-site.com', 'partner.org', 'resource.net', 'example.co'];
const domain = domains[Math.floor(Math.random() * domains.length)];
url = `https://${domain}/link-${Math.floor(Math.random() * 100)}`;
} else {
const extensions = ['.pdf', '.doc', '.jpg', '.zip'];
const ext = extensions[Math.floor(Math.random() * extensions.length)];
url = `https://${website}/files/document${Math.floor(Math.random() * 100)}${ext}`;
}

links.push({
url: url,
status: selectedStatus.status,
type: selectedStatus.type,
foundOn: pages[Math.floor(Math.random() * pages.length)]
});
}

return links;
}

function displayResults(links, website) {
// Calculate statistics
const broken = links.filter(link => link.type === 'broken').length;
const redirects = links.filter(link => link.type === 'redirect').length;
const working = links.filter(link => link.type === 'working').length;
const total = links.length;

// Update stats
brokenCount.textContent = broken;
redirectCount.textContent = redirects;
workingCount.textContent = working;
totalCount.textContent = total;

// Display links (show broken first, then redirects, then working)
const sortedLinks = [...links].sort((a, b) => {
const order = { broken: 1, redirect: 2, working: 3 };
return order[a.type] - order[b.type];
});

sortedLinks.forEach(link => {
const linkItem = document.createElement('div');
linkItem.className = `link-item ${link.type}`;

let statusText = '';
let statusClass = '';

switch(link.type) {
case 'broken':
statusText = `‚ùå ${link.status} Error`;
statusClass = 'status-broken';
break;
case 'redirect':
statusText = `‚Ü™Ô∏è ${link.status} Redirect`;
statusClass = 'status-redirect';
break;
case 'working':
statusText = `‚úÖ ${link.status} OK`;
statusClass = 'status-working';
break;
}

let icon = '';
switch(link.type) {
case 'broken': icon = 'fas fa-unlink'; break;
case 'redirect': icon = 'fas fa-random'; break;
case 'working': icon = 'fas fa-link'; break;
}

linkItem.innerHTML = `
<div class="link-status ${statusClass}">${statusText}</div>
<div class="link-url">
<i class="${icon}"></i>
<a href="${link.url}" target="_blank">${link.url}</a>
</div>
<div class="link-details">
<i class="fas fa-map-marker-alt"></i> Found on: ${link.foundOn} |
<i class="fas fa-clock"></i> Last checked: Just now
</div>
<div class="link-actions">
<button class="action-btn fix" onclick="visitLink('${link.url}')">
<i class="fas fa-external-link-alt"></i> Visit Link
</button>
<button class="action-btn fix" onclick="copyLink('${link.url}')">
<i class="fas fa-copy"></i> Copy URL
</button>
${link.type === 'broken' ? `
<button class="action-btn fix" onclick="suggestFix('${link.url}')">
<i class="fas fa-wrench"></i> Suggest Fix
</button>
` : ''}
</div>
`;

linksContainer.appendChild(linkItem);
});
}

function downloadReport() {

csvContent += `Scan Date: ${date}\r\n`;
csvContent += "URL,Status,Type,Found On\r\n";

links.forEach(link => {
const url = link.querySelector('.link-url a').textContent;
const status = link.querySelector('.link-status').textContent;

link.classList.contains('redirect') ? 'Redirect' : 'Working';

csvContent += `"${url}","${status}","${type}","${foundOn}"\r\n`;
});

const encodedUri = encodeURI(csvContent);
const link = document.createElement("a");
link.setAttribute("href", encodedUri);
link.setAttribute("download", `broken-links-report-${website}-${date}.csv`);
document.body.appendChild(link);
