<script>
let currentDevice = 'mobile';

function setDevice(device) {
currentDevice = device;
document.querySelectorAll('.device-btn').forEach(btn => btn.classList.remove('active'));
event.target.classList.add('active');
if (document.getElementById('resultsSection').style.display === 'block') {
analyzeSpeed();
}
}

function analyzeSpeed() {
const url = document.getElementById('websiteUrl').value.trim();
if (!url) {
alert('Please enter a URL');
return;
}

document.getElementById('resultsSection').style.display = 'block';

// Simulate API call
setTimeout(() => {
// Generate realistic scores
let lcp, fid, cls, score;

if (url.includes('google') || url.includes('facebook')) {
lcp = (1.5 + Math.random()).toFixed(1);
fid = Math.floor(30 + Math.random() * 20);
cls = (0.02 + Math.random() * 0.03).toFixed(2);
score = 90 + Math.floor(Math.random() * 8);
} else {
lcp = (2.0 + Math.random() * 2).toFixed(1);
fid = Math.floor(50 + Math.random() * 50);
cls = (0.05 + Math.random() * 0.1).toFixed(2);
score = 60 + Math.floor(Math.random() * 30);
}

// Update CWV
document.getElementById('lcpValue').innerHTML = lcp + 's';
document.getElementById('fidValue').innerHTML = fid + 'ms';
document.getElementById('clsValue').innerHTML = cls;
document.getElementById('performanceScore').innerHTML = score;
document.getElementById('scoreProgress').style.width = score + '%';

// Update status badges
updateStatus('lcp', parseFloat(lcp), [2.5, 4.0]);
updateStatus('fid', fid, [100, 300]);
updateStatus('cls', parseFloat(cls), [0.1, 0.25]);
updateScoreStatus(score);

// Generate detailed metrics
generateDetailedMetrics();

// Generate suggestions
generateSuggestions();
}, 1500);
}

function updateStatus(metric, value, thresholds) {
const element = document.getElementById(metric + 'Status');
const progressFill = document.getElementById(metric + 'Progress');

if (value < thresholds[0]) {
element.className = 'cww-status status-good';
element.innerHTML = 'Good';
} else if (value < thresholds[1]) {
element.className = 'cww-status status-average';
element.innerHTML = 'Needs Improvement';
} else {
element.className = 'cww-status status-poor';
element.innerHTML = 'Poor';
}
}

function updateScoreStatus(score) {
const element = document.getElementById('scoreStatus');
if (score >= 90) {
element.className = 'cww-status status-good';
element.innerHTML = 'Excellent';
} else if (score >= 50) {
element.className = 'cww-status status-average';
element.innerHTML = 'Average';
} else {
element.className = 'cww-status status-poor';
element.innerHTML = 'Poor';
}
}

function generateDetailedMetrics() {
const metrics = [
{ name: 'First Contentful Paint (FCP)', value: '1.8s', status: 'good' },
{ name: 'Time to Interactive (TTI)', value: '3.2s', status: 'average' },
{ name: 'Total Blocking Time (TBT)', value: '150ms', status: 'good' },
{ name: 'Speed Index', value: '2.1s', status: 'good' }
];

const container = document.getElementById('detailedMetrics');
container.innerHTML = metrics.map(m => `
<div class="metric-card">
<div class="metric-header">
<span class="metric-name">${m.name}</span>
<span class="metric-value">${m.value}</span>
</div>
<div class="progress-bar">
<div class="progress-fill fill-${m.status}" style="width: ${m.status === 'good' ? '80%' : '50%'}"></div>
</div>
</div>
`).join('');
}

function generateSuggestions() {
const suggestions = [
'Optimize images - Compress and serve in WebP format',
'Enable text compression - Use Gzip or Brotli',
'Leverage browser caching - Set appropriate cache headers',
'Minimize render-blocking resources - Defer non-critical CSS/JS',
'Reduce unused JavaScript - Remove unnecessary code'
];

const container = document.getElementById('suggestionsList');
container.innerHTML = suggestions.map(s => `
<div class="suggestion-item">
<i class="fas fa-check-circle" style="color: #10b981; margin-right: 10px;"></i>
${s}
</div>
`).join('');
}
