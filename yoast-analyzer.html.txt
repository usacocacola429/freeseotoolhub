<script>
function analyzeContent() {
const keyword = document.getElementById('focusKeyword').value.trim();
const content = document.getElementById('content').value.trim();

if (!keyword || !content) {
alert('Please enter both keyword and content');
return;
}

document.getElementById('resultsSection').style.display = 'block';

// Calculate scores
const keywordDensity = (content.toLowerCase().split(keyword.toLowerCase()).length - 1) / content.split(' ').length * 100;
const contentLength = content.split(' ').length;

let score = 70; // Base score

// Keyword in title (check first 10 words)
const firstWords = content.split(' ').slice(0, 10).join(' ').toLowerCase();
if (firstWords.includes(keyword.toLowerCase())) score += 10;

// Keyword density check (ideal 0.5-2.5%)
if (keywordDensity >= 0.5 && keywordDensity <= 2.5) score += 10;

// Content length check
if (contentLength >= 300) score += 10;
else if (contentLength < 150) score -= 10;

// Meta description length (simulated)
const metaDescLength = 155; // Simulated
if (metaDescLength >= 120 && metaDescLength <= 160) score += 5;

// Update score circle
const circle = document.getElementById('scoreCircle');
circle.innerHTML = Math.min(100, Math.max(0, score));

if (score >= 80) {
circle.className = 'score-circle score-good';
} else if (score >= 60) {
circle.className = 'score-circle score-average';
} else {
circle.className = 'score-circle score-poor';
}

// Generate analysis items
const analysis = [
{ title: 'Keyword in Title', status: firstWords.includes(keyword.toLowerCase()) ? 'good' : 'bad', desc: 'Focus keyword should appear in the first paragraph' },
{ title: 'Keyword Density', status: (keywordDensity >= 0.5 && keywordDensity <= 2.5) ? 'good' : 'bad', desc: `${keywordDensity.toFixed(2)}% (ideal: 0.5-2.5%)` },
{ title: 'Content Length', status: contentLength >= 300 ? 'good' : 'bad', desc: `${contentLength} words (minimum 300 recommended)` },
{ title: 'Meta Description', status: 'good', desc: 'Meta description length is good (155 characters)' },
{ title: 'Readability Score', status: contentLength > 200 ? 'good' : 'bad', desc: 'Content is easy to read' },
{ title: 'Outbound Links', status: 'bad', desc: 'Consider adding external links to authoritative sources' }
];

const container = document.getElementById('analysisResults');
container.innerHTML = analysis.map(a => `
<div class="analysis-item">
<div class="analysis-header">
<span class="analysis-title">${a.title}</span>
<span class="analysis-status status-${a.status === 'good' ? 'good' : 'bad'}">
${a.status === 'good' ? '‚úÖ Good' : '‚ùå Needs improvement'}
</span>
</div>
<p style="color: #64748b; font-size: 14px;">${a.desc}</p>
</div>
`).join('');
}
</script>
