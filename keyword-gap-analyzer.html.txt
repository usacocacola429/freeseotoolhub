{ keyword: 'cheap seo services india', volume: 1900, difficulty: 25, cpc: 2.5, category: 'india' },
{ keyword: 'हिंदी में एसईओ', volume: 1500, difficulty: 18, cpc: 0.8, category: 'hindi' },
{ keyword: 'local seo for small business', volume: 1700, difficulty: 32, cpc: 3.2, category: 'local' },
{ keyword: 'seo agency in bangalore', volume: 1950, difficulty: 30, cpc: 3.9, category: 'india' }
];

let allKeywords = [...baseKeywords];
if (market === 'india') {
allKeywords = [...allKeywords, ...indiaKeywords];
}

// Determine which keywords each domain ranks for
return allKeywords.map(kw => {
// Your domain rankings (some keywords)
const yourRank = Math.random() > 0.6 ? Math.floor(Math.random() * 30) + 1 : null;

// Competitor rankings
const comp1Rank = Math.random() > 0.2 ? Math.floor(Math.random() * 25) + 1 : null;
const comp2Rank = Math.random() > 0.3 ? Math.floor(Math.random() * 25) + 1 : null;

const competitorCount = [comp1Rank, comp2Rank].filter(r => r !== null).length;

// Determine status
let status = 'missing';
if (yourRank !== null && competitorCount > 0) {
const avgCompRank = ([comp1Rank, comp2Rank].filter(r => r !== null).reduce((a,b) => a + b, 0) / competitorCount);
status = yourRank < avgCompRank ? 'strong' : 'weak';
} else if (yourRank !== null) {
status = 'unique';
} else if (competitorCount > 0) {
status = 'missing';
}

// Calculate opportunity score
let opportunityScore = 0;
if (status === 'missing') {
opportunityScore = Math.min(100, (kw.volume / 200) + (100 - kw.difficulty) / 2);
} else if (status === 'weak') {
opportunityScore = Math.min(100, (kw.volume / 250) + (100 - kw.difficulty) / 3);
}

return {
...kw,
yourRank: yourRank,
comp1Rank: comp1Rank,
comp2Rank: comp2Rank,
competitorCount: competitorCount,
status: status,
opportunityScore: Math.round(opportunityScore),
priority: opportunityScore > 70 ? 'High' : (opportunityScore > 40 ? 'Medium' : 'Low')
};
});
}

// Update table with filtering
function updateTable(keywords, filter = 'missing') {
let filtered = keywords;
if (filter === 'missing') filtered = keywords.filter(k => k.status === 'missing');
else if (filter === 'weak') filtered = keywords.filter(k => k.status === 'weak');
else if (filter === 'strong') filtered = keywords.filter(k => k.status === 'strong');

// Sort by opportunity score
filtered.sort((a, b) => b.opportunityScore - a.opportunityScore);

let html = '';
filtered.slice(0, 50).forEach(k => {
const oppClass = k.opportunityScore > 70 ? 'opportunity-high' : (k.opportunityScore > 40 ? 'opportunity-medium' : 'opportunity-low');
const difficultyClass = k.difficulty < 30 ? 'text-success' : (k.difficulty < 60 ? 'text-warning' : 'text-danger');
const competitorText = [k.comp1Rank, k.comp2Rank].filter(r => r !== null).map(r => `#${r}`).join(', ') || 'None';

html += `
<tr>
<td><strong>${k.keyword}</strong></td>
<td>${k.volume.toLocaleString()}<div class="volume-bar"><div class="volume-fill" style="width: ${Math.min(100, k.volume / 200)}%"></div></div></td>
<td class="${difficultyClass}">${k.difficulty}</td>
<td>$${k.cpc}</td>
<td><small>${competitorText || '-'}</small></td>
<td><span class="opportunity-badge ${oppClass}">${k.opportunityScore}%</span></td>
<td><span class="badge ${k.priority === 'High' ? 'bg-danger' : (k.priority === 'Medium' ? 'bg-warning' : 'bg-secondary')}">${k.priority}</span></td>
</tr>
`;
});
tableBody.innerHTML = html;
}

// Export functions
function exportCSV(keywords) {
const headers = ['Keyword', 'Search Volume', 'Difficulty', 'CPC', 'Status', 'Opportunity Score', 'Priority'];
const rows = keywords.map(k => [
k.keyword, k.volume, k.difficulty, k.cpc, k.status, k.opportunityScore, k.priority
]);

const csvContent = [headers, ...rows].map(row => row.join(',')).join('\n');
const blob = new Blob([csvContent], { type: 'text/csv' });
const url = window.URL.createObjectURL(blob);
const a = document.createElement('a');
a.href = url;
a.download = `keyword-gap-analysis-${new Date().toISOString().split('T')[0]}.csv`;
a.click();
}
