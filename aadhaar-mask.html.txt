<script>
// DOM Elements
const aadhaarInput = document.getElementById('aadhaarInput');

const clearBtn = document.getElementById('clearBtn');

const resultBox = document.getElementById('resultBox');

const customDigitsContainer = document.getElementById('customDigitsContainer');

const bulkInput = document.getElementById('bulkInput');

// Mask function
function maskAadhaar() {
let aadhaar = aadhaarInput.value.trim();
if (!aadhaar) {
showMessage('Please enter an Aadhaar number!', 'error');
return;
}

// Remove spaces and non-digits
aadhaar = aadhaar.replace(/\D/g, '');

// Validate length
if (aadhaar.length !== 12) {
showMessage('Aadhaar must be 12 digits!', 'error');
return;
}

// Get selected mask style
const selectedStyle = document.querySelector('input[name="maskStyle"]:checked').value;
let masked = '';
let note = '';

switch(selectedStyle) {
case 'last4':
masked = 'XXXX XXXX ' + aadhaar.slice(-4);
note = 'âœ“ Only last 4 digits visible - Safe for sharing';
break;
case 'first2last4':
masked = aadhaar.slice(0, 2) + 'XX XXXX ' + aadhaar.slice(-4);
note = 'âœ“ First 2 and last 4 digits visible';
break;
case 'middle':
masked = 'XXXX ' + aadhaar.slice(4, 8) + ' ' + aadhaar.slice(-4);
note = 'âœ“ Middle 4 digits visible (bank format)';
break;
case 'custom':
const positions = customDigits.value.split(',').map(p => parseInt(p.trim())).filter(p => !isNaN(p) && p >= 1 && p <= 12);
let result = '';
for (let i = 0; i < 12; i++) {
if (positions.includes(i + 1)) {
result += aadhaar[i];
} else {
result += 'X';
}
if (i === 3 || i === 7) result += ' ';
}
masked = result;
note = `âœ“ Digits at positions ${positions.join(', ')} visible`;
break;
default:
masked = 'XXXX XXXX ' + aadhaar.slice(-4);
note = 'âœ“ Only last 4 digits visible';
}

// Format with spaces
if (!masked.includes(' ')) {
masked = masked.slice(0, 4) + ' ' + masked.slice(4, 8) + ' ' + masked.slice(8);
}

maskedResult.textContent = masked;
maskNote.textContent = note;
resultBox.style.display = 'block';

// Update counter
counter++;
localStorage.setItem('aadhaarMasked', counter);
processedCount.textContent = counter.toLocaleString();

showMessage('âœ… Aadhaar masked successfully!', 'success');
}

// Bulk mask function
function bulkMask() {
const bulkText = bulkInput.value.trim();
if (!bulkText) {
showMessage('Please enter Aadhaar numbers!', 'error');
return;
}

const lines = bulkText.split('\n');
const maskedLines = [];
const selectedStyle = document.querySelector('input[name="maskStyle"]:checked').value;

lines.forEach(line => {
line = line.trim();
if (!line) return;

// Extract digits
const digits = line.replace(/\D/g, '');
if (digits.length === 12) {
let masked = '';
switch(selectedStyle) {
case 'last4':
masked = 'XXXX XXXX ' + digits.slice(-4);
break;
case 'first2last4':
masked = digits.slice(0, 2) + 'XX XXXX ' + digits.slice(-4);
break;
case 'middle':
masked = 'XXXX ' + digits.slice(4, 8) + ' ' + digits.slice(-4);
break;
case 'custom':
const positions = customDigits.value.split(',').map(p => parseInt(p.trim())).filter(p => !isNaN(p) && p >= 1 && p <= 12);
let result = '';
for (let i = 0; i < 12; i++) {
if (positions.includes(i + 1)) {
result += digits[i];
} else {
result += 'X';
}
if (i === 3 || i === 7) result += ' ';
}
masked = result;
break;
}
maskedLines.push(masked);
} else {
maskedLines.push('âŒ Invalid: ' + line);
}
});

bulkResultText.innerHTML = maskedLines.join('<br>');
bulkResult.style.display = 'block';
counter += maskedLines.length;
localStorage.setItem('aadhaarMasked', counter);
processedCount.textContent = counter.toLocaleString();

showMessage(`âœ… ${maskedLines.length} Aadhaar numbers masked!`, 'success');
}
