// Grammar rules
const grammarRules = [
{
pattern: /\b(I|we|you|they|he|she|it) (is|am|are)\b/gi,
correction: 'Use appropriate verb form',
type: 'subject-verb agreement'
},
{
pattern: /\b(a|an) (apple|orange|umbrella|hour|honest)/gi,
correction: 'Use "an" before vowel sounds, "a" before consonant sounds',
type: 'article usage'
},
{
pattern: /(?<!\.)\s+([A-Z])/g,
correction: 'Start sentence with capital letter after period',
type: 'capitalization'
},
{
pattern: /\b(much|many) (people|books|water|information)/gi,
correction: 'Use "many" for countable nouns, "much" for uncountable',
type: 'countable/uncountable'
}
];

// Punctuation rules
const punctuationRules = [
{
pattern: /([a-z])([A-Z])/g,
correction: 'Missing period or space between sentences',
type: 'missing punctuation'
},
{
pattern: /\s+([.,!?])/g,
correction: 'Remove space before punctuation',
type: 'spacing'
},
{
pattern: /([.!?])([a-z])/g,
correction: 'Add space after punctuation',
type: 'spacing'
}
];

const errors = [];
const words = text.split(/\b/);
let corrected = text;

// Check spelling
if (spellCheck.checked) {
Object.keys(spellingMistakes).forEach(wrongWord => {
const regex = new RegExp(`\\b${wrongWord}\\b`, 'gi');
if (regex.test(text)) {
errors.push({
type: 'spelling',
wrong: wrongWord,
correct: spellingMistakes[wrongWord],
message: `Spelling mistake: "${wrongWord}" should be "${spellingMistakes[wrongWord]}"`
});
corrected = corrected.replace(regex, spellingMistakes[wrongWord]);
}
});
}

// Check grammar
if (grammarCheck.checked) {
grammarRules.forEach(rule => {
const matches = [...text.matchAll(rule.pattern)];
matches.forEach(match => {
errors.push({
type: 'grammar',
wrong: match[0],
correct: rule.correction,
message: `Grammar error: ${rule.type} - ${rule.correction}`
});
});
});
}

// Check punctuation
if (punctuationCheck.checked) {
punctuationRules.forEach(rule => {
const matches = [...text.matchAll(rule.pattern)];
matches.forEach(match => {
errors.push({
type: 'punctuation',
wrong: match[0],
correct: rule.correction,
message: `Punctuation: ${rule.type} - ${rule.correction}`
});
});
});
}

// Add style suggestions
const styleSuggestions = [
{
pattern: /\b(very|really|extremely)\s+(\w+)/gi,
suggestion: 'Use stronger word instead of "very" + adjective',
type: 'style'
},
{
pattern: /\b(a lot of|lots of)\b/gi,
suggestion: 'Consider using "many", "numerous", or "several"',
type: 'word choice'
},
{
pattern: /\.\s+But\b/gi,
suggestion: 'Consider starting new sentence instead of using "But"',
type: 'sentence structure'
}
];

const suggestions = [];
styleSuggestions.forEach(suggestion => {
const matches = [...text.matchAll(suggestion.pattern)];
matches.forEach(match => {
suggestions.push({
type: 'suggestion',
text: match[0],
suggestion: suggestion.suggestion,
category: suggestion.type
});
});
});

// Calculate scores
const totalWords = text.split(/\s+/).length;
const errorRate = (errors.length / totalWords) * 100;
const accuracy = Math.max(0, 100 - errorRate);
const overall = Math.round(accuracy);

return {
errors,
suggestions,
correctedText: corrected,
stats: {
totalWords,
errorCount: errors.length,
suggestionCount: suggestions.length,
accuracy,
overallScore: overall
}
};
}
